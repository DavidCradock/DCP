127 // std::vector<std::string> vStrImageFilenames; // Holds each individual image's filename

129 // std::vector<CImageAtlasDetails> _mvImageDetails; // Holds each individual image's details

136 {

139 {

141 // Assign i to temp
142 pImageTemp = vImages[i];
143 strFilenameTemp = vStrImageFilenames[i];
144 imageAtlasDetailsTemp = _mvImageDetails[i];

146 vImages[i] = vImages[j];
147 vStrImageFilenames[i] = vStrImageFilenames[j];
148 _mvImageDetails[i] = _mvImageDetails[j];
149 // Assign temp to j
150 vImages[j] = pImageTemp;
151 vStrImageFilenames[j] = strFilenameTemp;
152 _mvImageDetails[j] = imageAtlasDetailsTemp;
153 }
154 }
155 }
156
157 // Bubble sort temp vectors by their image's height
158 for (int i = 0; i < (int)vImages.size(); ++i)
159 {
160 for (int j = 0; j < (int)vImages.size(); ++j)
161 {
162 // Swap by largest first
163 if (_mvImageDetails[j].vDims.y < _mvImageDetails[i].vDims.y)
164 {
165 // Temporarily store image, filename and image details
166 // Assign i to temp
167 pImageTemp = vImages[i];
168 strFilenameTemp = vStrImageFilenames[i];
169 imageAtlasDetailsTemp = _mvImageDetails[i];
170 // Assign j to i
171 vImages[i] = vImages[j];
172 vStrImageFilenames[i] = vStrImageFilenames[j];
173 _mvImageDetails[i] = _mvImageDetails[j];
174 // Assign temp to j
175 vImages[j] = pImageTemp;
176 vStrImageFilenames[j] = strFilenameTemp;
177 _mvImageDetails[j] = imageAtlasDetailsTemp;
178 }
179 }
180 }
181 // Now, vStrImageFilenames, vImages and _mvImageDetails contain the filename and images sorted by greatest width, then height first
182
183 // Now we have to following arrays filled...
184 // std::vector<std::string> vStrImageFilenames; // Holds each individual image's filename
185 // std::vector<CImage*> vImages; // Holds each individual image's image data
186 // std::vector<CImageAtlasDetails> _mvImageDetails; // Holds each individual image's details
187 // _mvImageDetails[X].sTexCoords is not computed yet, we'll temporarily use this to hold position of each individual image inside of texture atlas
188 // and then convert to actual texture coordinates later further below.
189
190 // We now need to compute each image's position within the large atlas image and once done, we'll have the dimensions of the atlas image/s and the number required.
191 std::vector<CVector2f> vv2fAtlasDims; // Will hold computed dims of each atlas image
192 vv2fAtlasDims.push_back(CVector2f(0.0f, (float)vImages[0]->getHeight()));
193 int iCurAtlasImage = 0;
194 CVector2f v2fCurrentPositionInAtlas(0.0f, 0.0f);
195 float fSpacing = (float)uiImageSpacing;
196 float fSpacingTimesTwo = fSpacing * 2.0f;
197
198 // We go from left to right for each row, adding images one at a time, then when we can no longer add any more
199 // images along the X axis, we move down a row and add more images starting from the left again.
200 for (unsigned int ui = 0; ui < vImages.size(); ui++)
201 {
202 // Compute position of current image in atlas
203 CVector2f vImageDims = vImages[ui]->getDimensions();
204 _mvImageDetails[ui].sTexCoords.vTL.x = fSpacing + v2fCurrentPositionInAtlas.x;
205 _mvImageDetails[ui].sTexCoords.vTL.y = fSpacing + v2fCurrentPositionInAtlas.y;
206 _mvImageDetails[ui].sTexCoords.vBR.x = fSpacingTimesTwo + v2fCurrentPositionInAtlas.x + vImageDims.x;
207 _mvImageDetails[ui].sTexCoords.vBR.y = fSpacingTimesTwo + v2fCurrentPositionInAtlas.y + vImageDims.y;
208
209 // If the image position fits in current atlas row
210 if (_mvImageDetails[ui].sTexCoords.vBR.x <= (float)uiMaxAtlasImageWidth)
211 {
212 // Image fits, fill in the rest of it's information
213 _mvImageDetails[ui].sTexCoords.vTR.x = _mvImageDetails[ui].sTexCoords.vBR.x;
214 _mvImageDetails[ui].sTexCoords.vTR.y = _mvImageDetails[ui].sTexCoords.vTL.y;
215 _mvImageDetails[ui].sTexCoords.vBL.x = _mvImageDetails[ui].sTexCoords.vTL.x;
216 _mvImageDetails[ui].sTexCoords.vBL.y = _mvImageDetails[ui].sTexCoords.vBR.y;
217
218 // Update maximum atlas size
219 if (vv2fAtlasDims[iCurAtlasImage].x < _mvImageDetails[ui].sTexCoords.vBR.x)
220 vv2fAtlasDims[iCurAtlasImage].x = _mvImageDetails[ui].sTexCoords.vBR.x;
221 if (vv2fAtlasDims[iCurAtlasImage].y < _mvImageDetails[ui].sTexCoords.vBR.y)
222 vv2fAtlasDims[iCurAtlasImage].y = _mvImageDetails[ui].sTexCoords.vBR.y;
223
224 // Move position in atlas right
225 v2fCurrentPositionInAtlas.x += fSpacingTimesTwo + vImageDims.x;
226
227 // Set atlas image index for current image
228 _mvImageDetails[ui].uiAtlasImage = iCurAtlasImage;
229 }
230 else // This image doesn't fit along current row, so start a new one
231 {
232 // Move position in atlas down by max image height in row
233 v2fCurrentPositionInAtlas.x = 0.0f;
234 v2fCurrentPositionInAtlas.y = vv2fAtlasDims[iCurAtlasImage].y - fSpacing;
235
236 // Compute position of current image in atlas
237 _mvImageDetails[ui].sTexCoords.vTL.x = fSpacing + v2fCurrentPositionInAtlas.x;
238 _mvImageDetails[ui].sTexCoords.vTL.y = fSpacing + v2fCurrentPositionInAtlas.y;
239 _mvImageDetails[ui].sTexCoords.vBR.x = fSpacingTimesTwo + v2fCurrentPositionInAtlas.x + vImageDims.x;
240 _mvImageDetails[ui].sTexCoords.vBR.y = fSpacingTimesTwo + v2fCurrentPositionInAtlas.y + vImageDims.y;
241
242 // If the image position fits in current atlas
243 if (_mvImageDetails[ui].sTexCoords.vBR.y <= (float)uiMaxAtlasImageHeight)
244 {
245 // Image fits, fill in the rest of it's information
246 _mvImageDetails[ui].sTexCoords.vTR.x = _mvImageDetails[ui].sTexCoords.vBR.x;
247 _mvImageDetails[ui].sTexCoords.vTR.y = _mvImageDetails[ui].sTexCoords.vTL.y;
248 _mvImageDetails[ui].sTexCoords.vBL.x = _mvImageDetails[ui].sTexCoords.vTL.x;
249 _mvImageDetails[ui].sTexCoords.vBL.y = _mvImageDetails[ui].sTexCoords.vBR.y;
250
251 // Update maximum atlas size
252 if (vv2fAtlasDims[iCurAtlasImage].x < _mvImageDetails[ui].sTexCoords.vBR.x)
253 vv2fAtlasDims[iCurAtlasImage].x = _mvImageDetails[ui].sTexCoords.vBR.x;
254 if (vv2fAtlasDims[iCurAtlasImage].y < _mvImageDetails[ui].sTexCoords.vBR.y)
255 vv2fAtlasDims[iCurAtlasImage].y = _mvImageDetails[ui].sTexCoords.vBR.y;
256
257 // Move position in atlas right
258 v2fCurrentPositionInAtlas.x += fSpacingTimesTwo + vImageDims.x;
259
260 // Set atlas image index for current image
261 _mvImageDetails[ui].uiAtlasImage = iCurAtlasImage;
262 }
263 else // This image doesn't fit in this atlas and maximum dims of the atlas have been reached
264 {
265 iCurAtlasImage++;
266 vv2fAtlasDims.push_back(CVector2f(0.0f, (float)vImages[ui]->getHeight()));
267 v2fCurrentPositionInAtlas.setZero();
268
269 // Add image to new atlas image
270 _mvImageDetails[ui].sTexCoords.vTL.x = fSpacing + v2fCurrentPositionInAtlas.x;

272 _mvImageDetails[ui].sTexCoords.vBR.x = fSpacingTimesTwo + v2fCurrentPositionInAtlas.x + vImageDims.x;
273 _mvImageDetails[ui].sTexCoords.vBR.y = fSpacingTimesTwo + v2fCurrentPositionInAtlas.y + vImageDims.y;

275 _mvImageDetails[ui].sTexCoords.vTR.y = _mvImageDetails[ui].sTexCoords.vTL.y;
276 _mvImageDetails[ui].sTexCoords.vBL.x = _mvImageDetails[ui].sTexCoords.vTL.x;
277 _mvImageDetails[ui].sTexCoords.vBL.y = _mvImageDetails[ui].sTexCoords.vBR.y;
278
279 // Update maximum atlas size
280 if (vv2fAtlasDims[iCurAtlasImage].x < _mvImageDetails[ui].sTexCoords.vBR.x)
281 vv2fAtlasDims[iCurAtlasImage].x = _mvImageDetails[ui].sTexCoords.vBR.x;
282 if (vv2fAtlasDims[iCurAtlasImage].y < _mvImageDetails[ui].sTexCoords.vBR.y)
283 vv2fAtlasDims[iCurAtlasImage].y = _mvImageDetails[ui].sTexCoords.vBR.y;

290 }
291 }
292 }

294 // Now we get here...
295 // iCurAtlasImage holds number of atlas images needed
296 // vv2fAtlasDims[X] holds each atlas image's dimensions
297 // _mvImageDetails[X].sTexCoords holds position in atlas of each individual image
298
299 // Create each atlas image
300 for (unsigned int ui = 0; ui < (unsigned int)iCurAtlasImage + 1; ui++)
301 {
302 CImage* pNewImage = new CImage;

305 _mvAtlasImages.push_back(pNewImage);
306 }
307
308 // Now copy each image into atlas image
309 for (unsigned int ui = 0; ui < vImages.size(); ui++)
310 {
311 vImages[ui]->copyRectTo(*_mvAtlasImages[_mvImageDetails[ui].uiAtlasImage],
312 0, 0,
313 (int)vImages[ui]->getWidth(),
314 (int)vImages[ui]->getHeight(),
315 (int)_mvImageDetails[ui].sTexCoords.vTL.x,
316 (int)_mvImageDetails[ui].sTexCoords.vTL.y);
317
318 // Now compute texture coordinates
319 CVector2f vAtlasDims = _mvAtlasImages[_mvImageDetails[ui].uiAtlasImage]->getDimensions();
320 // Prevent divide by zero
321 ThrowIfTrue(vAtlasDims.x < 1.0f, "CImageAtlasPacker::createAtlasImages() failed. Atlas width less than 1.");
322 ThrowIfTrue(vAtlasDims.y < 1.0f, "CImageAtlasPacker::createAtlasImages() failed. Atlas height less than 1.");
323 CVector2f vAtlasDimsRecip(1.0f / vAtlasDims.x, 1.0f / vAtlasDims.y);
324
325 // Compute correct image positions, not including the spacing
326 // From above, top left position is correct for x and y.
327 // All others include spacing and need to be recalculated.
328 _mvImageDetails[ui].sTexCoords.vBR.x = _mvImageDetails[ui].sTexCoords.vTL.x + _mvImageDetails[ui].vDims.x;
329 _mvImageDetails[ui].sTexCoords.vBR.y = _mvImageDetails[ui].sTexCoords.vTL.y + _mvImageDetails[ui].vDims.y;
330 _mvImageDetails[ui].sTexCoords.vTR.x = _mvImageDetails[ui].sTexCoords.vTL.x + _mvImageDetails[ui].vDims.x;
331 _mvImageDetails[ui].sTexCoords.vTR.y = _mvImageDetails[ui].sTexCoords.vTL.y;
332 _mvImageDetails[ui].sTexCoords.vBL.x = _mvImageDetails[ui].sTexCoords.vTL.x;
333 _mvImageDetails[ui].sTexCoords.vBL.y = _mvImageDetails[ui].sTexCoords.vTL.y + _mvImageDetails[ui].vDims.y;
334
335 // Now all positions are correct
336
337 // Now convert from pixel position in image to 0.0f - 1.0f
338 _mvImageDetails[ui].sTexCoords.vBL.x *= vAtlasDimsRecip.x;
339 _mvImageDetails[ui].sTexCoords.vBL.y *= vAtlasDimsRecip.y;
340 _mvImageDetails[ui].sTexCoords.vBR.x *= vAtlasDimsRecip.x;
341 _mvImageDetails[ui].sTexCoords.vBR.y *= vAtlasDimsRecip.y;
342 _mvImageDetails[ui].sTexCoords.vTL.x *= vAtlasDimsRecip.x;
343 _mvImageDetails[ui].sTexCoords.vTL.y *= vAtlasDimsRecip.y;
344 _mvImageDetails[ui].sTexCoords.vTR.x *= vAtlasDimsRecip.x;
345 _mvImageDetails[ui].sTexCoords.vTR.y *= vAtlasDimsRecip.y;
346
347 // For nearest filtering this is all we have to do.
348 // However, with linear filtering, we need to offset the texture coordinates
349 // by half a texel so that everything looks great when using linear filtering.
350 _mvImageDetails[ui].sTexCoords.vBL.x += vAtlasDimsRecip.x;
351 _mvImageDetails[ui].sTexCoords.vBL.y -= vAtlasDimsRecip.y;
352 _mvImageDetails[ui].sTexCoords.vBR.x -= vAtlasDimsRecip.x;
353 _mvImageDetails[ui].sTexCoords.vBR.y -= vAtlasDimsRecip.y;
354 _mvImageDetails[ui].sTexCoords.vTL.x += vAtlasDimsRecip.x;
355 _mvImageDetails[ui].sTexCoords.vTL.y += vAtlasDimsRecip.y;
356 _mvImageDetails[ui].sTexCoords.vTR.x -= vAtlasDimsRecip.x;
357 _mvImageDetails[ui].sTexCoords.vTR.y += vAtlasDimsRecip.y;
358
359 }
360
361 // Now we're done, we setup the hashmap for fast lookup of image details by individual image names
362 for (unsigned int ui = 0; ui < _mvImageDetails.size(); ui++)
363 {
364 _mmapImageDetails[_mvImageDetails[ui].strImageFilename] = _mvImageDetails[ui];
365 }
366
367 // Free each individual loaded image
368 for (unsigned int ui = 0; ui < vImages.size(); ui++)
369 {
370 delete vImages[ui];
371 }
372 vImages.clear();
373 }
