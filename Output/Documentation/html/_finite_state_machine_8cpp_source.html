22 {
23 // Find named state
24 std::map<std::string, CFiniteStateBase*>::iterator it = _mmapStates.find(strStateName);
25 ThrowIfTrue(it == _mmapStates.end(), "CFiniteStateMachine::switchToState(\"" + strStateName + "\") failed. State name doesn't exist.");

27 // Find currently set state

29
30 // If the states are the same, do nothing.

33 return;

40 // Enter new state
41 it->second->onEnter();
42
43 // Set new state to currently set state
44 _mstrCurrentState = strStateName;
