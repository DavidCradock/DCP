180 {
181 float fAngleRadians = deg2rad(fAngleDegrees);
182 float fCos = cosf(fAngleRadians);
183 float fSin = sinf(fAngleRadians);
184 float fOMC = 1.0f - fCos;
185
186 m[0] = fCos + (vAxis.x * vAxis.x) * fOMC;
187 m[5] = fCos + (vAxis.y * vAxis.y) * fOMC;
188 m[10] = fCos + (vAxis.z * vAxis.z) * fOMC;
189 m[15] = 1.0f;

191 m[1] = vAxis.x * vAxis.y * fOMC - vAxis.z * fSin;

193 m[2] = vAxis.x * vAxis.z * fOMC - vAxis.y * fSin;
194 m[9] = vAxis.y * vAxis.z * fOMC + vAxis.x * fSin;

308 void CMatrix::setProjectionPerspective(
309 float fWidth,
310 float fHeight,
311 float fFieldOfViewInDegrees,
312 float fNearClippingPlaneDistance,
313 float fFarClippingPlaneDistance,
314 float fAspectRatio)
315 {
316 float fFOVRadians = deg2rad(fFieldOfViewInDegrees);
317 float fTanHalfFOV = tan(fFOVRadians * 0.5f);
318 float fNearMinusFar = fNearClippingPlaneDistance - fFarClippingPlaneDistance;
319 ThrowIfTrue(fNearMinusFar == 0.0f, "CMatrix::setProjectionPerspective() failed. Given near and far clipping plane distances invalid.");
320
321 // Compute fAspectRatio if given a value <= 0.0f from the back buffer's framebuffer's resource dimensions
322 if (fAspectRatio <= 0.0f)
323 {
324 fAspectRatio = fWidth / fHeight;
325 }
326
327 // First column
328 m[0] = 1.0f / (fAspectRatio * fTanHalfFOV);
329 m[1] = 0.0f;
330 m[2] = 0.0f;
331 m[3] = 0.0f;
332
333 // Second column
334 m[4] = 0.0f;
335 m[5] = 1.0f / fTanHalfFOV;
336 m[6] = 0.0f;
337 m[7] = 0.0f;
338
339 // Third column
340 m[8] = 0.0f;
341 m[9] = 0.0f;
342 m[10] = fFarClippingPlaneDistance / (fNearMinusFar);
343 m[11] = -1.0f;
344
345 // Fourth column
346 m[12] = 0.0f;
347 m[13] = 0.0f;
348 m[14] = -(fFarClippingPlaneDistance * fNearClippingPlaneDistance) / (fFarClippingPlaneDistance - fNearClippingPlaneDistance);
349 m[15] = 1.0f; // 0.0f?
350 }

352 void CMatrix::setProjectionOrthographic(
353 float fWidth,
354 float fHeight,
355 float fLeft,
356 float fRight,
357 float fTop,
358 float fBottom,
359 float fNear,
360 float fFar)
361 {
362 // Compute fRight and fBottom if given a value <= 0.0f
363 if (fRight <= 0.0f || fBottom <= 0.0f)
364 {
365 fRight = fWidth;
366 fBottom = fHeight;
367 }
368
369 // First column
370 m[0] = 2.0f / (fRight - fLeft);
371 m[1] = 0.0f;
372 m[2] = 0.0f;
373 m[3] = 0.0f;
374
375 // Second column
376 m[4] = 0.0f;
377 m[5] = 2.0f / (fTop - fBottom);
378 m[6] = 0.0f;
379 m[7] = 0.0f;
380
381 // Third column
382 m[8] = 0.0f;
383 m[9] = 0.0f;
384 m[10] = -1.0f / (fFar - fNear);
385 m[11] = 0.0f;
386
387 // Fourth column
388 m[12] = -(fRight + fLeft) / (fRight - fLeft);
389 m[13] = -(fTop + fBottom) / (fTop - fBottom);
390 m[14] = -fNear / (fFar - fNear);
391 m[15] = 1.0f;
392 }

394 void CMatrix::setViewLookat(const CVector3f& vEyePos, const CVector3f& vTarget, const CVector3f& vUp)
395 {
396 CVector3f vf = vTarget - vEyePos; vf.normalise();
397 CVector3f vs = vf.getCross(vUp); vs.normalise();
398 CVector3f vu = vs.getCross(vf); vu.normalise(); // No need to normalise as other two vectors already are.
399
400 // First column
401 m[0] = vs.x;
402 m[1] = vu.x;
403 m[2] = -vf.x;
404 m[3] = 0.0f;
405
406 // Second column
407 m[4] = vs.y;
408 m[5] = vu.y;
409 m[6] = -vf.y;
410 m[7] = 0.0f;
411
412 // Third column
413 m[8] = vs.z;
414 m[9] = vu.z;
415 m[10] = -vf.z;
416 m[11] = 0.0f;
417
418 // Fourth column
419 m[12] = -vs.getDot(vEyePos);
420 m[13] = -vu.getDot(vEyePos);
421 m[14] = vf.getDot(vEyePos);
422 m[15] = 1.0f;
423 }
