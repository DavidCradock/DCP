398 inline void CImage::setPixel(int iX, int iY, unsigned char r, unsigned char g, unsigned char b, unsigned char a)
399 {
400 if (iX >= _miWidth)
401 return;
402 if (iY >= _miHeight)
403 return;
404
405 unsigned int iIndex = iX + (iY * _miWidth);
406 iIndex *= _miNumChannels;
407 switch (_miNumChannels)
408 {
409 case 1:
410 _mpData[iIndex] = r;
411 break;
412 case 3:
413 _mpData[iIndex] = r;
414 _mpData[iIndex + 1] = g;
415 _mpData[iIndex + 2] = b;
416 break;
417 case 4:
418 _mpData[iIndex] = r;
419 _mpData[iIndex + 1] = g;
420 _mpData[iIndex + 2] = b;
421 _mpData[iIndex + 3] = a;
422 break;
423 }

454 inline bool CImage::_isPixelEdge(int iPosX, int iPosY, unsigned char r, unsigned char g, unsigned char b)
455 {
456 // Don't check edge pixels of image
457 if (iPosX == 0)
458 return false;
459 if (iPosY == 0)
460 return false;
461 if (iPosX >= int(_miWidth) - 1)
462 return false;
463 if (iPosY >= int(_miHeight) - 1)
464 return false;
465
466 // Get center pixel colour values
467 unsigned char col[4];
468 getPixel(iPosX, iPosY, col[0], col[1], col[2], col[3]);
469
470 // If the center pixel's colour is same as mask, it's not an edge
471 if (col[0] == r)
472 if (col[1] == g)
473 if (col[2] == b)
474 return false;
475
476 // If we get here, the center pixel is not the same as the mask colour
477
478 // Get bordering pixels of center pixel
479 // If they are the same as the mask, then it's an edge
480
481 // Left
482 getPixel(iPosX - 1, iPosY, col[0], col[1], col[2], col[3]);
483 if (col[0] == r)
484 if (col[1] == g)
485 if (col[2] == b)
486 return true;
487 // Right
488 getPixel(iPosX + 1, iPosY, col[0], col[1], col[2], col[3]);
489 if (col[0] == r)
490 if (col[1] == g)
491 if (col[2] == b)
492 return true;
493
494 // Top
495 getPixel(iPosX, iPosY - 1, col[0], col[1], col[2], col[3]);
496 if (col[0] == r)
497 if (col[1] == g)
498 if (col[2] == b)
499 return true;
500
501 // Bottom
502 getPixel(iPosX, iPosY + 1, col[0], col[1], col[2], col[3]);
503 if (col[0] == r)
504 if (col[1] == g)
505 if (col[2] == b)
506 return true;
507
508 // Top left
509 getPixel(iPosX - 1, iPosY - 1, col[0], col[1], col[2], col[3]);
510 if (col[0] == r)
511 if (col[1] == g)
512 if (col[2] == b)
513 return true;
514
515 // Top right
516 getPixel(iPosX + 1, iPosY - 1, col[0], col[1], col[2], col[3]);
517 if (col[0] == r)
518 if (col[1] == g)
519 if (col[2] == b)
520 return true;
521
522 // Bottom left
523 getPixel(iPosX - 1, iPosY + 1, col[0], col[1], col[2], col[3]);
524 if (col[0] == r)
525 if (col[1] == g)
526 if (col[2] == b)
527 return true;
528
529 // Bottom right
530 getPixel(iPosX + 1, iPosY + 1, col[0], col[1], col[2], col[3]);
531 if (col[0] == r)
532 if (col[1] == g)
533 if (col[2] == b)
534 return true;
535
536 return false; // Shouldn't get here, keep compiler happy
537 }
