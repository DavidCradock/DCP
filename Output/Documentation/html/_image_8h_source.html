397 inline void CImage::setPixel(int iX, int iY, unsigned char r, unsigned char g, unsigned char b, unsigned char a)
398 {
399 if (iX >= _miWidth)
400 return;
401 if (iY >= _miHeight)
402 return;
403
404 unsigned int iIndex = iX + (iY * _miWidth);
405 iIndex *= _miNumChannels;
406 switch (_miNumChannels)
407 {
408 case 1:
409 _mpData[iIndex] = r;
410 break;
411 case 3:
412 _mpData[iIndex] = r;
413 _mpData[iIndex + 1] = g;
414 _mpData[iIndex + 2] = b;
415 break;
416 case 4:
417 _mpData[iIndex] = r;
418 _mpData[iIndex + 1] = g;
419 _mpData[iIndex + 2] = b;
420 _mpData[iIndex + 3] = a;
421 break;

453 inline bool CImage::_isPixelEdge(int iPosX, int iPosY, unsigned char r, unsigned char g, unsigned char b)
454 {
455 // Don't check edge pixels of image
456 if (iPosX == 0)
457 return false;
458 if (iPosY == 0)
459 return false;
460 if (iPosX >= int(_miWidth) - 1)
461 return false;
462 if (iPosY >= int(_miHeight) - 1)
463 return false;
464
465 // Get center pixel colour values
466 unsigned char col[4];
467 getPixel(iPosX, iPosY, col[0], col[1], col[2], col[3]);
468
469 // If the center pixel's colour is same as mask, it's not an edge
470 if (col[0] == r)
471 if (col[1] == g)
472 if (col[2] == b)
473 return false;
474
475 // If we get here, the center pixel is not the same as the mask colour
476
477 // Get bordering pixels of center pixel
478 // If they are the same as the mask, then it's an edge
479
480 // Left
481 getPixel(iPosX - 1, iPosY, col[0], col[1], col[2], col[3]);
482 if (col[0] == r)
483 if (col[1] == g)
484 if (col[2] == b)
485 return true;
486 // Right
487 getPixel(iPosX + 1, iPosY, col[0], col[1], col[2], col[3]);
488 if (col[0] == r)
489 if (col[1] == g)
490 if (col[2] == b)
491 return true;
492
493 // Top
494 getPixel(iPosX, iPosY - 1, col[0], col[1], col[2], col[3]);
495 if (col[0] == r)
496 if (col[1] == g)
497 if (col[2] == b)
498 return true;
499
500 // Bottom
501 getPixel(iPosX, iPosY + 1, col[0], col[1], col[2], col[3]);
502 if (col[0] == r)
503 if (col[1] == g)
504 if (col[2] == b)
505 return true;
506
507 // Top left
508 getPixel(iPosX - 1, iPosY - 1, col[0], col[1], col[2], col[3]);
509 if (col[0] == r)
510 if (col[1] == g)
511 if (col[2] == b)
512 return true;
513
514 // Top right
515 getPixel(iPosX + 1, iPosY - 1, col[0], col[1], col[2], col[3]);
516 if (col[0] == r)
517 if (col[1] == g)
518 if (col[2] == b)
519 return true;
520
521 // Bottom left
522 getPixel(iPosX - 1, iPosY + 1, col[0], col[1], col[2], col[3]);
523 if (col[0] == r)
524 if (col[1] == g)
525 if (col[2] == b)
526 return true;
527
528 // Bottom right
529 getPixel(iPosX + 1, iPosY + 1, col[0], col[1], col[2], col[3]);
530 if (col[0] == r)
531 if (col[1] == g)
532 if (col[2] == b)
533 return true;
534
535 return false; // Shouldn't get here, keep compiler happy
536 }
