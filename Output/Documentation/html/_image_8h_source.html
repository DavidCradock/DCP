392 inline void CImage::setPixel(int iX, int iY, unsigned char r, unsigned char g, unsigned char b, unsigned char a)
393 {
394 if (iX >= _miWidth)
395 return;
396 if (iY >= _miHeight)
397 return;
398
399 unsigned int iIndex = iX + (iY * _miWidth);
400 iIndex *= _miNumChannels;
401 switch (_miNumChannels)
402 {
403 case 1:
404 _mpData[iIndex] = r;
405 break;
406 case 3:
407 _mpData[iIndex] = r;

448 inline bool CImage::_isPixelEdge(int iPosX, int iPosY, unsigned char r, unsigned char g, unsigned char b)
449 {
450 // Don't check edge pixels of image
451 if (iPosX == 0)
452 return false;
453 if (iPosY == 0)
454 return false;
455 if (iPosX >= int(_miWidth) - 1)
456 return false;
457 if (iPosY >= int(_miHeight) - 1)
458 return false;
459
460 // Get center pixel colour values
461 unsigned char col[4];
462 getPixel(iPosX, iPosY, col[0], col[1], col[2], col[3]);
463
464 // If the center pixel's colour is same as mask, it's not an edge
465 if (col[0] == r)
466 if (col[1] == g)
467 if (col[2] == b)
468 return false;
469
470 // If we get here, the center pixel is not the same as the mask colour
471
472 // Get bordering pixels of center pixel
473 // If they are the same as the mask, then it's an edge
474
475 // Left
476 getPixel(iPosX - 1, iPosY, col[0], col[1], col[2], col[3]);
477 if (col[0] == r)
478 if (col[1] == g)
479 if (col[2] == b)
480 return true;
481 // Right
482 getPixel(iPosX + 1, iPosY, col[0], col[1], col[2], col[3]);
483 if (col[0] == r)
484 if (col[1] == g)
485 if (col[2] == b)
486 return true;
487
488 // Top
489 getPixel(iPosX, iPosY - 1, col[0], col[1], col[2], col[3]);
490 if (col[0] == r)
491 if (col[1] == g)
492 if (col[2] == b)
493 return true;
494
495 // Bottom
496 getPixel(iPosX, iPosY + 1, col[0], col[1], col[2], col[3]);
497 if (col[0] == r)
498 if (col[1] == g)
499 if (col[2] == b)
500 return true;
501
502 // Top left
503 getPixel(iPosX - 1, iPosY - 1, col[0], col[1], col[2], col[3]);
504 if (col[0] == r)
505 if (col[1] == g)
506 if (col[2] == b)
507 return true;
508
509 // Top right
510 getPixel(iPosX + 1, iPosY - 1, col[0], col[1], col[2], col[3]);
511 if (col[0] == r)
512 if (col[1] == g)
513 if (col[2] == b)
514 return true;
515
516 // Bottom left
517 getPixel(iPosX - 1, iPosY + 1, col[0], col[1], col[2], col[3]);
518 if (col[0] == r)
519 if (col[1] == g)
520 if (col[2] == b)
521 return true;
522
523 // Bottom right
524 getPixel(iPosX + 1, iPosY + 1, col[0], col[1], col[2], col[3]);
525 if (col[0] == r)
526 if (col[1] == g)
527 if (col[2] == b)
528 return true;
529
530 return false; // Shouldn't get here, keep compiler happy
531 }
