39 {
40 // Check for valid parameters

44 ThrowIfTrue(iNumNeuronsPerLayer < 1 && iNumLayers > 0, "CNeuralNetworkFeedForward::create() given invalid number of neurons per layer. Must be at least 1 if iNumLayers is > 0");
45
46 // Clear everything
47 _mvecLayers.clear();
48
49 // Set given parameters
50 _miNumInputs = iNumInputs;
51 _miNumOutputs = iNumOutputs;
52 _miNumLayers = iNumLayers;
53 _miNumNeuronsPerLayer = iNumNeuronsPerLayer;
54
55 // Set default bias
56 _mdBias = -1;
57
58 // Set default sigmoid response
59 _mdSigmoidResponse = 1;
60
61 if (_miNumLayers)
62 {
63 // Create the first layer of neurons which have the number of inputs set to the number of inputs of the network.
64 _mvecLayers.push_back(CNeuronLayer(_miNumNeuronsPerLayer, _miNumInputs));
65
66 // Create the other layers
67 // iNumLayers -1 because we've already created a layer above
68 for (int i = 0; i < _miNumLayers - 1; i++)
69 {
70 // Create the layer, using the number of inputs set to the number of neurons per layer
71 _mvecLayers.push_back(CNeuronLayer(_miNumNeuronsPerLayer, _miNumNeuronsPerLayer));
72 }
73
74 // Create the output layer
75 _mvecLayers.push_back(CNeuronLayer(_miNumOutputs, _miNumNeuronsPerLayer));
76 }
77 else
78 {
79 // Create the output layer
80 _mvecLayers.push_back(CNeuronLayer(_miNumOutputs, _miNumInputs));
81 }

85 {

87
88 // For each layer (+1 as we are also returning the output layer)
89 for (int iLayer = 0; iLayer < _miNumLayers + 1; iLayer++)
90 {
91 // For each neuron in each layer
92 for (int iNeuron = 0; iNeuron < _mvecLayers[iLayer].miNumNeurons; iNeuron++)

142 std::vector<double> CNeuralNetworkFeedForward::update(std::vector<double>& vecInputs)
143 {
144 ThrowIfTrue(vecInputs.size() != _miNumInputs, "CNeuralNetworkFeedForward::update() failed. Was given a vector of inputs which are not equal to in size of the network's number of inputs");
145 std::vector<double> vecOutputs;
146
147 int cWeight = 0;
148
149 // For each layer (+1 as we are also updating the output layer)
150 for (int iLayer = 0; iLayer < _miNumLayers + 1; iLayer++)
151 {
152 // If we've used the inputs, we need to set the inputs to be the outputs of the previous layer
153 if (iLayer > 0)
154 vecInputs = vecOutputs;
155
156 // Clear the outputs, as we're going to calculate them below
157 vecOutputs.clear();
158
159 // Used to count through each weight in the inputs below.
160 cWeight = 0;
161
162 int iNumInputs = 0;
163
164 // For each neuron in each layer
165 // multiply the input by the weight for that input, then use the sigmoid
166 // function to smooth it out to get the final output.
167 for (int iNeuron = 0; iNeuron < _mvecLayers[iLayer].miNumNeurons; iNeuron++)
168 {
169 double dAccumulatedInput = 0;
170 iNumInputs = _mvecLayers[iLayer].mvecNeurons[iNeuron].miNumInputs;
171
172 // For each weight, not including the bias (the -1)
173 for (int iWeight = 0; iWeight < iNumInputs - 1; iWeight++)
174 {
175 dAccumulatedInput += _mvecLayers[iLayer].mvecNeurons[iNeuron].mvecInputWeights[iWeight] * vecInputs[cWeight++];
176 }
177
178 // Multiply by the bias
179 dAccumulatedInput += _mvecLayers[iLayer].mvecNeurons[iNeuron].mvecInputWeights[iNumInputs - 1] * _mdBias;
180
181 // Pass the final computed input value, multiplied by the weights, with the bias applied, to the sigmoid function to
182 // obtain the final output value and store that in the vector of outputs.
183 vecOutputs.push_back(sigmoid(dAccumulatedInput, _mdSigmoidResponse));
184
185 cWeight = 0;
186 }
187 }
188 return vecOutputs;
189 }
