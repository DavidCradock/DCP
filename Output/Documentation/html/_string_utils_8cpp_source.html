125 std::string addFilenameExtension(const std::string& strFilenameExtension, const std::string& strFilename)
126 {
127 // Make sure valid values given.
128 ThrowIfTrue(0 == strFilenameExtension.length(), "CStringHelper::addFilenameExtension() failed. Given extension name of zero length.");
129 ThrowIfTrue(0 == strFilename.length(), "CStringHelper::addFilenameExtension() failed. Given file name of zero length.");
130
131 // Append "." to extension if needed
132 std::string strExt = strFilenameExtension;
133 if (strExt.c_str()[0] != '.')
134 {
135 std::string::iterator itBegin = strExt.begin();
136 strExt.insert(itBegin, '.');
137 }
138 std::string strFile = strFilename;
139 // Find last position of "." in given file name and remove everything after it
140 auto const pos = strFile.find_last_of('.');
141 if (pos != std::string::npos) // If "." found
142 {
143 // Remove "." and all following text
144 strFile.erase(pos, strFile.length() - pos);
145 }
146 // Append extension to filename
147 strFile.append(strExt);
148
149 // Make all lowercase
150 stringToLowercase(strFile);
151 return strFile;
152 }

217 std::vector<std::string> splitString(const std::string& string, const std::string& strSplitChars)
218 {
219 std::vector<std::string> out;
220
221 // If no strSplitChars found, simply add the entire string and return the result
222 size_t pos = string.find(strSplitChars, 0);
223 if (std::string::npos == pos)
224 {
225 out.push_back(string);
226 return out;
227 }
228
229 // If we get here, strSplitChars has been found in the string
230 std::string strLine;
231 std::string strAll = string;
232 while (std::string::npos != pos)
233 {
234 // Copy character upto the position of the found strSplitChars into strLine
235 strLine.assign(strAll, 0, pos);
236
237 // Add the line to the output
238 out.push_back(strLine);
239
240 // Reset strLine
241 strLine.clear();
242
243 // Remove all characters including the strSplitChars from strAll
244 strAll.erase(0, pos + strSplitChars.length());
245
246 // Find next position of strSplitChars in strAll
247 pos = strAll.find(strSplitChars, 0);
248 }
249 // If strAll still contains characters, add them to the vector
250 if (strAll.length())
251 {
252 out.push_back(strAll);
253 }
254
255 return out;
256 }
