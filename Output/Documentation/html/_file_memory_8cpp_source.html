42 // Get file size
43 std::ifstream ifstream(strFilename, std::ios::binary | std::ios::ate);
44 ThrowIfFalse(ifstream.is_open(), "CFileMemory::setFromFile(\"" + strFilename + "\") failed. Given filename could not be opened.");
