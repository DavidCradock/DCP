31 bool convertFileToHeader(const std::string &strFilename, const std::string &strArrayName, unsigned int uiNumElementsPerRow)
32 {
33 FILE* fs = NULL; // Source file
34 FILE* fd = NULL; // Destination file
35
36 std::string strFilenameInput = strFilename;
37 std::string strFilenameOutput = strFilename;
38 strFilenameOutput.append(".h");
39
40 // Open source
41 fopen_s(&fs, strFilename.c_str(), "rb");
42 if (fs == NULL)
43 return false;
44
45 // Open destination
46 fopen_s(&fd, strFilenameOutput.c_str(), "wb");
47 if (fd == NULL)
48 {
49 fclose(fs);
50 return false;
51 }
52
53 // Write comment giving name of binary file that this data came from
54 fprintf(fd, "// Following data taken from file \"%s\"\n", strFilename.c_str());
55
56 // Get length of data
57 fseek(fs, 0, SEEK_END);
58 int length = ftell(fs);
59 fseek(fs, 0, SEEK_SET);
60
61 // Write #define for length of array
62 std::string strArrayNameUppercase = strArrayName;
63 std::transform(strArrayNameUppercase.begin(), strArrayNameUppercase.end(), strArrayNameUppercase.begin(), ::toupper);
64 strArrayNameUppercase.append("_SIZE");
65 fprintf(fd, "#define %s %d\n", strArrayNameUppercase.c_str(), length);
66
67 // Write start of array
68 fprintf(fd, "unsigned char %s[] =\n", strArrayName.c_str());
69 fprintf(fd, "{\n ");
70
71 // Write out data from source binary file
72 while (feof(fs) == false)
73 {
74 unsigned char tmp;
75 for (int e = 0; e < (int)uiNumElementsPerRow; e++)
76 {
77 fread(&tmp, 1, sizeof(unsigned char), fs);
78 if (feof(fs) == false)
79 fprintf(fd, "0x%02x,", tmp);
80 }
81
82 if (feof(fs)) // Reached end of file
83 {
84 fseek(fd, -1, SEEK_CUR); // Remove last comma
85 fprintf(fd, "};");
86 }
87 fprintf(fd, "\n ");
88 }
89 fprintf(fd, "\n");
90
91 fclose(fs);
92 fclose(fd);
93 return true;
94 }

96 void getHueColour(float fHueAmount, float& fRed, float& fGreen, float& fBlue)
97 {
98 clamp(fHueAmount, 0.0f, 1.0f);
99 fHueAmount *= 360.0f;
100 if (fHueAmount <= 60.0f) // Inc green
101 {
102 fRed = 1.0f;
103 fGreen = fHueAmount / 60.0f;
104 fBlue = 0.0f;
105 }
106 else if (fHueAmount <= 120.0f) // Dec red
107 {
108 fRed = 1.0f - ((fHueAmount - 60.0f) / 60.0f);
109 fGreen = 1.0f;
110 fBlue = 0.0f;
111 }
112 else if (fHueAmount <= 180.0f) // Inc blue
113 {
114 fRed = 0.0f;
115 fGreen = 1.0f;
116 fBlue = (fHueAmount - 120.0f) / 60.0f;
117 }
118 else if (fHueAmount <= 240.0f) // Dec green
119 {
120 fRed = 0.0f;
121 fGreen = 1.0f - ((fHueAmount - 180.0f) / 60.0f);
122 fBlue = 1.0f;
123 }
124 else if (fHueAmount <= 300.0f) // Inc red
125 {
126 fRed = (fHueAmount - 240.0f) / 60.0f;
127 fGreen = 0.0f;
128 fBlue = 1.0f;
129 }
130 else // dec blue
131 {
132 fRed = 1.0f;
133 fGreen = 0.0f;
134 fBlue = 1.0f - ((fHueAmount - 300.0f) / 60.0f);
135 }

164 void getMemInfo(SMemInfo& memInfo)
165 {
166 SecureZeroMemory(&memInfo, sizeof(SMemInfo));
167
168 HANDLE hAppProcess = GetCurrentProcess();
169 if (!hAppProcess) // Found application process handle?
170 return;
171
172 // Retreive and store process mem usage
173 PROCESS_MEMORY_COUNTERS_EX pmc;
174 if (GetProcessMemoryInfo(hAppProcess, (PPROCESS_MEMORY_COUNTERS)&pmc, sizeof(PROCESS_MEMORY_COUNTERS_EX)))
175 {
176 memInfo.proc.iPageFaultCount = (unsigned int)pmc.PageFaultCount;
177 memInfo.proc.iPeakWorkingSetSize = (unsigned int)pmc.PeakWorkingSetSize;
