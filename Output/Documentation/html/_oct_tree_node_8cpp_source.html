155 CAABB COctTreeNode::computeChildNodeRegion(ChildNode childNode) const
156 {
157 // Compute dimensions of new child node using the region of this parent node.
158 CVector3f vChildDims = _mRegion.getHalfDimensions();
159 CVector3f vParentPosition = _mRegion.getPosition();
160 CVector3f vHalfChildDims = vChildDims * 0.5f;
161
162 // Calculate region of the child node
163 CAABB childNodeRegion;
164 CVector3f vChildPosition = vParentPosition;
165 switch (childNode)
166 {
167 case ChildNode::NEGX_NEGY_NEGZ:
168 vChildPosition.x -= vHalfChildDims.x;
169 vChildPosition.y -= vHalfChildDims.y;
170 vChildPosition.z -= vHalfChildDims.z;
171 childNodeRegion.setPosDims(vChildPosition, vChildDims);
172 break;
173 case ChildNode::NEGX_POSY_NEGZ:
174 vChildPosition.x -= vHalfChildDims.x;
175 vChildPosition.y += vHalfChildDims.y;
176 vChildPosition.z -= vHalfChildDims.z;
177 childNodeRegion.setPosDims(vChildPosition, vChildDims);
178 break;
179 case ChildNode::NEGX_NEGY_POSZ:
180 vChildPosition.x -= vHalfChildDims.x;
181 vChildPosition.y -= vHalfChildDims.y;
182 vChildPosition.z += vHalfChildDims.z;
183 childNodeRegion.setPosDims(vChildPosition, vChildDims);
184 break;
185 case ChildNode::NEGX_POSY_POSZ:
186 vChildPosition.x -= vHalfChildDims.x;
187 vChildPosition.y += vHalfChildDims.y;
188 vChildPosition.z += vHalfChildDims.z;
189 childNodeRegion.setPosDims(vChildPosition, vChildDims);
190 break;
191 case ChildNode::POSX_NEGY_NEGZ:
192 vChildPosition.x += vHalfChildDims.x;
193 vChildPosition.y -= vHalfChildDims.y;
194 vChildPosition.z -= vHalfChildDims.z;
195 childNodeRegion.setPosDims(vChildPosition, vChildDims);
196 break;
197 case ChildNode::POSX_POSY_NEGZ:
198 vChildPosition.x += vHalfChildDims.x;
199 vChildPosition.y += vHalfChildDims.y;
200 vChildPosition.z -= vHalfChildDims.z;
201 childNodeRegion.setPosDims(vChildPosition, vChildDims);
202 break;
203 case ChildNode::POSX_NEGY_POSZ:
204 vChildPosition.x += vHalfChildDims.x;
205 vChildPosition.y -= vHalfChildDims.y;
206 vChildPosition.z += vHalfChildDims.z;
207 childNodeRegion.setPosDims(vChildPosition, vChildDims);
208 break;
209 case ChildNode::POSX_POSY_POSZ:
210 vChildPosition.x += vHalfChildDims.x;
211 vChildPosition.y += vHalfChildDims.y;
212 vChildPosition.z += vHalfChildDims.z;
213 childNodeRegion.setPosDims(vChildPosition, vChildDims);
214 break;
215 default:
216 ThrowIfTrue(1, "COctTreeNode::computeChildNodeRegion() given invalid ChildNode to compute it's region.");
217 }
218 return childNodeRegion;

221 void COctTreeNode::addEntity(COctTreeEntity* pEntity)
222 {
223 // If this node has no children, attempt to add the entity to this node
224 if (!hasAnyChildNodes())
225 {
226 // We haven't reached max capacity for this node
227 // OR we've reached maximum node depth with this node
228 if (_mmapEntities.size() < _mpOctTree->_miMaxEntitiesPerNode ||
229 _muiNodeDepth == _mpOctTree->_muiMaxNodeDepth)
230 {
231 // Add the entity to this node
232 // No need to check if the new entity name already exists, as COctTree::addEntity() has already checked
233 _mmapEntities[pEntity->_mstrName] = pEntity;
234 pEntity->_mpNodeOwner = this; // Set node owner for the entity
235 return;
236 }
237 // If we've reached maximum capacity for this node and max node depth hasn't been reached
238 if (_mmapEntities.size() == _mpOctTree->_miMaxEntitiesPerNode)
239 {
240 // We need to create child node/s then move all the entities from this node into the children,
241 // as well as the new entity
242
243 // Compute potential child node regions
244 CAABB aabbNNN = computeChildNodeRegion(ChildNode::NEGX_NEGY_NEGZ);
245 CAABB aabbNNP = computeChildNodeRegion(ChildNode::NEGX_NEGY_POSZ);
246 CAABB aabbNPN = computeChildNodeRegion(ChildNode::NEGX_POSY_NEGZ);
247 CAABB aabbNPP = computeChildNodeRegion(ChildNode::NEGX_POSY_POSZ);
248 CAABB aabbPNN = computeChildNodeRegion(ChildNode::POSX_NEGY_NEGZ);
249 CAABB aabbPNP = computeChildNodeRegion(ChildNode::POSX_NEGY_POSZ);
250 CAABB aabbPPN = computeChildNodeRegion(ChildNode::POSX_POSY_NEGZ);
251 CAABB aabbPPP = computeChildNodeRegion(ChildNode::POSX_POSY_POSZ);
252
253 // Add the new entity to this node's entity map, so it'll be moved to the correct child node below
254 _mmapEntities[pEntity->_mstrName] = pEntity;
255 pEntity->_mpNodeOwner = this; // Set node owner for the entity
256
257 // Move all the entities from this node, into the child nodes
258 std::map<std::string, COctTreeEntity*>::iterator it = _mmapEntities.begin();
259 while (it != _mmapEntities.end())
260 {
261 // Determine which child node the entity fits in, regardless of whether the child node exists or not
262 ChildNode childNode = ChildNode::NONE;
263 if (aabbNNN.getPointIsInside(it->second->_mvPosition))
264 childNode = ChildNode::NEGX_NEGY_NEGZ;
265 else if (aabbNNP.getPointIsInside(it->second->_mvPosition))
266 childNode = ChildNode::NEGX_NEGY_POSZ;
267 else if (aabbNPN.getPointIsInside(it->second->_mvPosition))
268 childNode = ChildNode::NEGX_POSY_NEGZ;
269 else if (aabbNPP.getPointIsInside(it->second->_mvPosition))
270 childNode = ChildNode::NEGX_POSY_POSZ;
271 else if (aabbPNN.getPointIsInside(it->second->_mvPosition))
272 childNode = ChildNode::POSX_NEGY_NEGZ;
273 else if (aabbPNP.getPointIsInside(it->second->_mvPosition))
274 childNode = ChildNode::POSX_NEGY_POSZ;
275 else if (aabbPPN.getPointIsInside(it->second->_mvPosition))
276 childNode = ChildNode::POSX_POSY_NEGZ;
277 else if (aabbPPP.getPointIsInside(it->second->_mvPosition))
278 childNode = ChildNode::POSX_POSY_POSZ;
279
280 // Error checking, making sure the entity could fit in one of the eight possible children
281 ThrowIfTrue(ChildNode::NONE == childNode, "COctTreeNode::addEntity() failed when trying to add entity " + it->second->_mstrName + " to any of the eight child nodes as it's position doesn't fit inside any of them.");
282
283 // Create the child node if it doesn't exist
284 if (!_mpChildNode[childNode])
285 {
286 _mpChildNode[childNode] = new COctTreeNode(computeChildNodeRegion(childNode), this, _mpOctTree);
287 ThrowIfFalse(_mpChildNode[childNode], "COctTreeNode::addEntity() failed when trying to add entity " + it->second->_mstrName + " to any of the eight child nodes as it was unable to allocate memory for the child node.");
288 }
289 _mpChildNode[childNode]->addEntity(it->second);
290 it++;
291 }
292 // Remove all entities from this node as they are now stored in the children
293 _mmapEntities.clear();
294 return;
295 }
296 }
297
298 // If we get here, then this node has children, so add the new entity to one of those...
299 // Determine which child node the entity fits in
300 // Compute potential child node regions
301 CAABB aabbNNN = computeChildNodeRegion(ChildNode::NEGX_NEGY_NEGZ);
302 CAABB aabbNNP = computeChildNodeRegion(ChildNode::NEGX_NEGY_POSZ);
303 CAABB aabbNPN = computeChildNodeRegion(ChildNode::NEGX_POSY_NEGZ);
304 CAABB aabbNPP = computeChildNodeRegion(ChildNode::NEGX_POSY_POSZ);
305 CAABB aabbPNN = computeChildNodeRegion(ChildNode::POSX_NEGY_NEGZ);
306 CAABB aabbPNP = computeChildNodeRegion(ChildNode::POSX_NEGY_POSZ);
307 CAABB aabbPPN = computeChildNodeRegion(ChildNode::POSX_POSY_NEGZ);
308 CAABB aabbPPP = computeChildNodeRegion(ChildNode::POSX_POSY_POSZ);
309
310 ChildNode childNode = ChildNode::NONE;
311 if (aabbNNN.getPointIsInside(pEntity->_mvPosition))
312 childNode = ChildNode::NEGX_NEGY_NEGZ;
313 else if (aabbNNP.getPointIsInside(pEntity->_mvPosition))
314 childNode = ChildNode::NEGX_NEGY_POSZ;
315 else if (aabbNPN.getPointIsInside(pEntity->_mvPosition))
316 childNode = ChildNode::NEGX_POSY_NEGZ;
317 else if (aabbNPP.getPointIsInside(pEntity->_mvPosition))
318 childNode = ChildNode::NEGX_POSY_POSZ;
319 else if (aabbPNN.getPointIsInside(pEntity->_mvPosition))
320 childNode = ChildNode::POSX_NEGY_NEGZ;
321 else if (aabbPNP.getPointIsInside(pEntity->_mvPosition))
322 childNode = ChildNode::POSX_NEGY_POSZ;
323 else if (aabbPPN.getPointIsInside(pEntity->_mvPosition))
324 childNode = ChildNode::POSX_POSY_NEGZ;
325 else if (aabbPPP.getPointIsInside(pEntity->_mvPosition))
326 childNode = ChildNode::POSX_POSY_POSZ;
327
328 // Error checking, making sure the entity could fit in one of the eight possible children
329 ThrowIfTrue(ChildNode::NONE == childNode, "COctTreeNode::addEntity() failed when trying to add entity " + pEntity->_mstrName + " to any of the eight child nodes as it's position doesn't fit inside any of them.");
330
331 // Create the child node if it doesn't exist
332 if (!_mpChildNode[childNode])
333 {
334 _mpChildNode[childNode] = new COctTreeNode(computeChildNodeRegion(childNode), this, _mpOctTree);
335 ThrowIfFalse(_mpChildNode[childNode], "COctTreeNode::addEntity() failed when trying to add entity " + pEntity->_mstrName + " to any of the eight child nodes as it was unable to allocate memory for the child node.");
336 }
337
338 // Add the new entity
339 _mpChildNode[childNode]->addEntity(pEntity);
340 }
