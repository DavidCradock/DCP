176 float CVector2f::getAngleDegreesSigned(const CVector2f& vOther) const
177 {
178 // Normalise this vector
179 CVector2f vThisNorm(x, y);
180 vThisNorm.normalise();
181
182 // Normalise other vector
183 CVector2f vOtherNorm(vOther.x, vOther.y);
184 vOtherNorm.normalise();
185
186 // Compute dot
187 float fTmp = vThisNorm.x * vOtherNorm.x + vThisNorm.y * vOtherNorm.y;
188
189 // Convert to unsigned radians
190 // make sure fTmp is valid
191 if (fTmp < -1.0f)
192 fTmp = -1.0f;
193 else if (fTmp > 1.0f)
194 fTmp = 1.0f;
195 fTmp = (float)acosf(fTmp);
196
197 // Convert to signed radians
198 if (vThisNorm.y * vOtherNorm.x < vThisNorm.x * vOtherNorm.y)
199 fTmp *= -1.0f;
200
201 // Convert to degrees
202 // NOTE: 180.0f / Pi is used to convert radians to degrees
203 // and it is a value of : 57.2957795
204 return fTmp * 57.2957795f;
205 }

207 float CVector2f::getAngleDegrees360(void) const
208 {
209 // Normalise this vector
210 CVector2f vThisNorm(x, y);
211 vThisNorm.normalise();
212
213 CVector2f vNorth(0.0f, 1.0f);
214
215 // Compute dot
216 float fTmp = vThisNorm.x * vNorth.x + vThisNorm.y * vNorth.y;
217
218 // Convert to unsigned radians
219 // First, make sure fTmp is valid
220 if (fTmp < -1.0f)
221 fTmp = -1.0f;
222 else if (fTmp > 1.0f)
223 fTmp = 1.0f;
224 fTmp = (float)acosf(fTmp);
225
226 // Convert to degrees
227 // NOTE: 180.0f / Pi is used to convert radians to degrees
228 // and it is a value of : 57.2957795
229 fTmp *= 57.2957795f;
230
231 if (vThisNorm.y * vNorth.x < vThisNorm.x * vNorth.y)

233 return fTmp;
234 }
235 return 180.0f + (180.0f - fTmp);

238 void CVector2f::rotate(float fDegrees)
239 {

241 float fOldX = x;
242 float fOldY = y;
243 float fSin, fCos;
244 fSin = sinf(fRad);
245 fCos = cosf(fRad);
246 x = fOldX * fCos - fOldY * fSin;
247 y = fOldX * fSin + fOldY * fCos;
