38 CGenome CGeneticsTraining::rouletteWheelSelection(const std::vector<CGenome>& vecPopulation) const
39 {
40 // Make sure the vector contains stuff
41 ThrowIfFalse(vecPopulation.size(), "CGeneticsTraining::rouletteWheelSelection() failed as the given CGenome vector was empty.");
42
43 // Compute total fitness of all the elements in the vector
44 double dTotalFitness = 0;
45 for (unsigned int ui = 0; ui < vecPopulation.size(); ui++)
46 {
47 dTotalFitness += vecPopulation[ui].dFitness;
48 }
49
50 // Now compute a random value between 0 and dTotalFitness.
51 // Think of this as a portion of the entire wheel.
52 double dPortion = randDouble() * dTotalFitness;
53
54 // Now go through the vector of genomes, adding their fitness scores as we go
55 // and if the sub total is greater than the above dTotalFitness,
56 // we've found our index.
57 double dSubTotal = 0;
58 CGenome selectedGenome = vecPopulation[0];
59 for (unsigned int ui = 0; ui < vecPopulation.size(); ui++)
60 {
61 dSubTotal += vecPopulation[ui].dFitness;
62 if (dSubTotal > dPortion)
63 {
64 selectedGenome = vecPopulation[ui];
65 break;
66 }
67 }
68 return selectedGenome;
69 }

72 {
73 dWorstFitness = 999999999999999;
74 dBestFitness = -999999999999999;
75 dTotalFitness = 0;
76 for (unsigned int ui = 0; ui < vecPopulationFitness.size(); ui++)
77 {
78 // Total fitness
79 dTotalFitness += vecPopulationFitness[ui];
80
81 // Worst fitness
82 if (dWorstFitness > vecPopulationFitness[ui])
83 dWorstFitness = vecPopulationFitness[ui];
84
85 // Best fitness
86 if (dBestFitness < vecPopulationFitness[ui])
87 dBestFitness = vecPopulationFitness[ui];

119 void CGeneticsTraining::crossover(
120 std::vector<double>& vecMumWeights,
121 std::vector<double>& vecDadWeights,
122 std::vector<double>& vecChildAWeights,
123 std::vector<double>& vecChildBWeights,
124 double dCrossoverRate) const
125 {
126 // Error checking to make sure vector sizes are equal
127 if (vecMumWeights.size() != vecDadWeights.size())
128 Throw("CGeneticsTraining::crossover() failed. The passed parent vectors are not the same size.");
129
130 // Determine whether we should perform the crossover...
131 // First by calculating a random value an determining whether it is below the given crossover rate.
132 // And second, if both parents are the same
133 if (randDouble() > dCrossoverRate || vecMumWeights == vecDadWeights)
134 {
135 vecChildAWeights = vecMumWeights;
136 vecChildBWeights = vecDadWeights;
137 return;
138 }
139
140 // If we get here, mum and dad are different and a crossover is to be performed.
141
142 // Determine a random cross over point
143 int iCrossoverPoint = randInt(0, (int)vecMumWeights.size() - 1);
144
145 // Clear the child weights
146 vecChildAWeights.clear();
147 vecChildBWeights.clear();
148
149 // Give the children weights from mum and dad, before the crossover point
150 for (int i = 0; i < iCrossoverPoint; i++)
151 {
152 // Child A gets all of mum's weights, before the crossover point
153 vecChildAWeights.push_back(vecMumWeights[i]);
154
155 // The other child, all of dad's, before the crossover point
156 vecChildBWeights.push_back(vecDadWeights[i]);
157 }
158 // Give the children weights from dad and mum, after the crossover point
159 for (int i = iCrossoverPoint; i < vecMumWeights.size(); i++)
160 {
161 // Child A gets all of dad's weights, after the crossover point
162 vecChildAWeights.push_back(vecDadWeights[i]);
163
164 // The other child, all of mum's, after the crossover point
165 vecChildBWeights.push_back(vecMumWeights[i]);
166 }
167 }

169 void CGeneticsTraining::crossoverBetweenNeurons(
170 std::vector<double>& vecMumWeights,
171 std::vector<double>& vecDadWeights,
172 std::vector<double>& vecChildAWeights,
173 std::vector<double>& vecChildBWeights,
174 const std::vector<int>& vecSplitPoints,
175 double dCrossoverRate) const
176 {
177 // Error checking to make sure vector sizes are equal
178 if (vecMumWeights.size() != vecDadWeights.size())
179 Throw("CGeneticsTraining::crossoverBetweenNeurons() failed. The passed parent vectors are not the same size.");
180 // Error checking to make sure the given split points vector is of adequate size
181 ThrowIfTrue(vecSplitPoints.size() < 2, "CGeneticsTraining::crossoverBetweenNeurons() failed. The passed vecSplitPoints vector's size is too small. Must be at least 2.");
182
183 // Determine whether we should perform the crossover...
184 // First by calculating a random value an determining whether it is below the given crossover rate.
185 // And second, if both parents are the same
186 if (randDouble() > dCrossoverRate || vecMumWeights == vecDadWeights)
187 {
188 // Just pass on the parents as the children.
189 vecChildAWeights = vecMumWeights;
190 vecChildBWeights = vecDadWeights;
191 return;
192 }
193
194 // If we get here, mum and dad are different and a crossover is to be performed.
195
196 // Using the vector holding where we are allowed to split the weights, so that they do not split a portion of a single neuron,
197 // determine two crossover points
198 int iRand1 = randInt(0, (int)vecSplitPoints.size() - 2);
199 int iCrossoverPoint1 = vecSplitPoints[iRand1];
200 int iRand2 = randInt(iRand1+1, (int)vecSplitPoints.size() - 1);
201 int iCrossoverPoint2 = vecSplitPoints[iRand2];
202
203 // Clear the child weights
204 vecChildAWeights.clear();
205 vecChildBWeights.clear();
206
207 // Give the children weights from mum and dad
208 for (int i = 0; i < (int)vecMumWeights.size(); i++)
209 {
210 // If outside of the crossover points
211 if (i < iCrossoverPoint1 || i > iCrossoverPoint2)
212 {
213 // Child A gets all of mum's weights
214 vecChildAWeights.push_back(vecMumWeights[i]);
215
216 // The other child, all of dad's
217 vecChildBWeights.push_back(vecDadWeights[i]);
218 }
219 else
220 {
221 // Child A gets all of dad's weights
222 vecChildAWeights.push_back(vecDadWeights[i]);
223
224 // The other child, all of mum's
225 vecChildBWeights.push_back(vecMumWeights[i]);
226 }
227 }

244 {
245 // Sort the old population by fitness so that the fitest are towards the end of the vector
246 std::sort(vecOldPop.begin(), vecOldPop.end());
247
248 // For each number of elite to insert
249 for (int i = 0; i < iNumberOfElite; i++)
250 {
251 // Add X copies of each elite
252 for (int j = 0; j < iNumCopiesOfEachElite; j++)
253 {
254 // Make sure there is room for the elite

267 std::vector<CGenome> CGeneticsTraining::createNewGeneration(
268 const std::vector<CGenome>& genomes,
269 const std::vector<int>& vecSplitsPoints,
270 double dMutationProbability,
271 double dMutationMaxAmount,
272 double dCrossoverRate,
273 int iNumberOfElite,
274 int iNumCopiesOfEachElite) const
275 {
276 // Error checking
277 ThrowIfFalse(genomes.size(), "CGeneticsTraining::createNewGeneration() failed, as it was given a vector of CGenome objects of zero size.");
278 ThrowIfTrue(iNumberOfElite < 0 || iNumCopiesOfEachElite < 0, "CGeneticsTraining::createNewGeneration() failed, as it was given values less than zero for iNumberOfElite or iNumCopiesOfEachElite.");
279
280 // The old population of genomes
281 std::vector<CGenome> vecOldPop = genomes;
282
283 // Compute worst, average, best and total fitness of the entire population
284 double dWorstFitness, dAverageFitness, dBestFitness, dTotalFitness;
285 computeFitness(vecOldPop, dWorstFitness, dAverageFitness, dBestFitness, dTotalFitness);
286
287 // Create the new population of genomes
288 std::vector<CGenome> vecNewPop;
289
290 // Insert the elite
291 insertElite(vecOldPop, vecNewPop, iNumberOfElite, iNumCopiesOfEachElite);
292
293 while (vecNewPop.size() < vecOldPop.size())
294 {
295 // Use roulette wheel selection to choose a mum and dad
296 CGenome mum = rouletteWheelSelection(vecOldPop);
297 CGenome dad = rouletteWheelSelection(vecOldPop);
298
299 // Perform some crossover to create the babies
300 CGenome baby1;
301 CGenome baby2;
302 crossoverBetweenNeurons(mum.vecWeights, dad.vecWeights, baby1.vecWeights, baby2.vecWeights, vecSplitsPoints, dCrossoverRate);
303
304 // Now mutate the babies (Yeuck!)
305 mutate(baby1.vecWeights, dMutationProbability, dMutationMaxAmount);
306 mutate(baby2.vecWeights, dMutationProbability, dMutationMaxAmount);
307
308 // Now add the new babies into the new population of genomes
309 vecNewPop.push_back(baby1);
310
311 // We check the new population size as the population size may be an odd number and if we add two
312 // babies, we'll create an oversized population.
313 if (vecNewPop.size() < vecOldPop.size())
314 vecNewPop.push_back(baby2);
315 }
316
317 // Error checking
318 return vecNewPop;
319 }
