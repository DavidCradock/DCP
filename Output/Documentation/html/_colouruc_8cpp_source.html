26 void CColouruc::setFromHSB(float fHueAmount, float fSaturation, float fBrightness)
27 {
28 // Clamp each passed value
29 clamp(fHueAmount, 0.0f, 1.0f);
30 clamp(fSaturation, 0.0f, 1.0f);
31 clamp(fBrightness, 0.0f, 1.0f);
32
33 // Compute colour
34 fHueAmount *= 360.0f;
35 float fReciprical60 = 1.0f / 60.0f;
36 float fRed, fGreen, fBlue;
37 if (fHueAmount <= 60.0f) // Inc green
38 {
39 fRed = 1.0f;
40 fGreen = fHueAmount * fReciprical60;
41 fBlue = 0.0f;
42 }
43 else if (fHueAmount <= 120.0f) // Dec red
44 {
45 fRed = 1.0f - ((fHueAmount - 60.0f) * fReciprical60);
46 fGreen = 1.0f;
47 fBlue = 0.0f;
48 }
49 else if (fHueAmount <= 180.0f) // Inc blue
50 {
51 fRed = 0.0f;
52 fGreen = 1.0f;
53 fBlue = (fHueAmount - 120.0f) * fReciprical60;
54 }
55 else if (fHueAmount <= 240.0f) // Dec green
56 {
57 fRed = 0.0f;
58 fGreen = 1.0f - ((fHueAmount - 180.0f) * fReciprical60);
59 fBlue = 1.0f;
60 }
61 else if (fHueAmount <= 300.0f) // Inc red
62 {
63 fRed = (fHueAmount - 240.0f) * fReciprical60;
64 fGreen = 0.0f;
65 fBlue = 1.0f;
66 }
67 else // dec blue
68 {
69 fRed = 1.0f;
70 fGreen = 0.0f;
71 fBlue = 1.0f - ((fHueAmount - 300.0f) * fReciprical60);
72 }
73
74 // Now RGB is set, apply saturation
75 float fSaturationScaleR = 1.0f - fRed;
76 float fSaturationScaleG = 1.0f - fGreen;
77 float fSaturationScaleB = 1.0f - fBlue;
78 fRed += fSaturationScaleR * fSaturation;
79 fGreen += fSaturationScaleG * fSaturation;
80 fBlue += fSaturationScaleB * fSaturation;
81 clamp(fRed, 0.0f, 1.0f);
82 clamp(fGreen, 0.0f, 1.0f);
83 clamp(fBlue, 0.0f, 1.0f);
84
85 // Now saturation is added, apply brightness
86 float fBrightnessInv = 1.0f - fBrightness;
87 fRed -= fBrightnessInv;
88 fGreen -= fBrightnessInv;
89 fBlue -= fBrightnessInv;
90 clamp(fRed, 0.0f, 1.0f);
91 clamp(fGreen, 0.0f, 1.0f);
92 clamp(fBlue, 0.0f, 1.0f);
93
94 // Convert float to unsigned char
95 red = unsigned char(fRed * 255);
96 green = unsigned char(fGreen * 255);
97 blue = unsigned char(fBlue * 255);
98 }

100 void CColouruc::getHSB(float& fHue, float& fSaturation, float& fBrightness) const
101 {
102 // Convert unsigned char to float
103 float fOneOver255 = 1.0f / 255.0f;
104 float fRed = float(red) * fOneOver255;
105 float fGreen = float(green) * fOneOver255;
106 float fBlue = float(blue) * fOneOver255;
107
108 // Get maximum and minimum values of current RGB values
109 float cmax = std::max(fRed, std::max(fGreen, fBlue));
110 float cmin = std::min(fRed, std::min(fGreen, fBlue));
111 float diff = cmax - cmin; // diff of cmax and cmin.
112 fHue = -1, fSaturation = -1;
113
114 // If cmax and cmax are equal then fHue = 0
115 if (cmax == cmin)
116 fHue = 0;
117
118 // If cmax equal r then compute fHue
119 else if (cmax == fRed)
120 fHue = float(fmod(60 * ((fGreen - fBlue) / diff) + 360, 360));
121
122 // If cmax equal g then compute fHue
123 else if (cmax == green)
124 fHue = float(fmod(60 * ((fBlue - fRed) / diff) + 120, 360));
125
126 // If cmax equal b then compute fHue
127 else if (cmax == blue)
128 fHue = float(fmod(60 * ((fRed - fGreen) / diff) + 240, 360));
129
130 fHue /= 360.0f;
131
132 // If cmax equal zero
133 if (cmax == 0)
134 fSaturation = 0;
135 else
136 fSaturation = (diff / cmax);// *100;
137
138 // Compute fBrightness
139 fBrightness = cmax;// *100;
140 }
