51 void CLog::add(const std::string& strText, const std::string& strFunctionName, const std::string& strLineNumber, const std::string& strSourceFilename)
52 {
53 _mQueueMutex.lock();
54
55 CLogEntry entry;
56 entry.strText = strText;
57 entry.strFunctionName = strFunctionName;
58 entry.strSourceFilename = strSourceFilename;
59 entry.strSourceLineNumber = strLineNumber;
60
61 // Compute time related stuff
62 _timer.update();
63 _timer.getClock(entry.fTimeSeconds, entry.iTimeMin, entry.iTimeHours, entry.iTimeDays, entry.iTimeWeeks);
64
65 // Create the current runtime as a string in the log entry
66 entry.strTime += std::to_string(entry.iTimeWeeks) + "w:";
67 entry.strTime += std::to_string(entry.iTimeDays) + "d:";
68 entry.strTime += std::to_string(entry.iTimeHours) + "h:";
69 entry.strTime += std::to_string(entry.iTimeMin) + "m:";
70 if (entry.fTimeSeconds < 10.0f)
71 entry.strTime += "0";
72 entry.strTime += std::to_string(float(entry.fTimeSeconds)) + "s";
73
74 // Add entry to the queue
75 _mQueueEntriesToAdd.push(entry);
76
77 _mQueueMutex.unlock();
78 }

80 void CLog::_mainThreadLoop(void)
81 {
82 bool bLogToSave = false;
83 CLogEntry logToSave;
84
85 while (!_mStop) // This gets set to true in the destructor
86 {
87 // Determine whether there's at least one log entry in the queue
88 // If so, copy it ready to save to file.
89 // We perform the minimal of stuff here to help prevent locking the mutex which could slow down calls to add()
90 bLogToSave = false;
91 _mQueueMutex.lock();
92 if (!_mQueueEntriesToAdd.empty())
93 {
94 bLogToSave = true;
95 logToSave = _mQueueEntriesToAdd.front();
96 _mQueueEntriesToAdd.pop();
97 }
98 _mQueueMutex.unlock();
99
100 // If there is a log entry, save it to file
101 if (bLogToSave)
102 {
103 std::ofstream file(_mstrFilename, std::ios::app);
104 if (file.is_open())
105 {
106 file << _strTableColumnText[0] << logToSave.strTime;
107 file << _strTableColumnText[1] << logToSave.strText;
108 file << _strTableColumnText[2] << logToSave.strFunctionName;
109 file << _strTableColumnText[3] << logToSave.strSourceLineNumber;
110 file << _strTableColumnText[4] << logToSave.strSourceFilename;
111 file << "</div></td></tr>\n";
112 }
113 else
114 {
115 }
116 }
117 else
118 {
119 // If there wasn't a log entry to save, let's sleep for a bit.
120 // Not sleeping due to there being a log entry makes it so that on the next
121 // loop is fast and allows us to prevent entries from queueing up.
122 std::this_thread::sleep_for(std::chrono::milliseconds(100));
123 }
124 }
