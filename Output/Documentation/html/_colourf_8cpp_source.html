30 void CColourf::setFromHSB(float fHueAmount, float fSaturation, float fBrightness)
31 {
32 // Clamp each passed value
33 clamp(fHueAmount, 0.0f, 1.0f);
34 clamp(fSaturation, 0.0f, 1.0f);
35 clamp(fBrightness, 0.0f, 1.0f);
36
37 // Compute colour
38 fHueAmount *= 360.0f;
39 float fReciprical60 = 1.0f / 60.0f;
40 if (fHueAmount <= 60.0f) // Inc green
41 {
42 red = 1.0f;
43 green = fHueAmount * fReciprical60;
44 blue = 0.0f;
45 }
46 else if (fHueAmount <= 120.0f) // Dec red
47 {
48 red = 1.0f - ((fHueAmount - 60.0f) * fReciprical60);
49 green = 1.0f;
50 blue = 0.0f;
51 }
52 else if (fHueAmount <= 180.0f) // Inc blue
53 {
54 red = 0.0f;
55 green = 1.0f;
56 blue = (fHueAmount - 120.0f) * fReciprical60;
57 }
58 else if (fHueAmount <= 240.0f) // Dec green
59 {
60 red = 0.0f;
61 green = 1.0f - ((fHueAmount - 180.0f) * fReciprical60);
62 blue = 1.0f;
63 }
64 else if (fHueAmount <= 300.0f) // Inc red
65 {
66 red = (fHueAmount - 240.0f) * fReciprical60;
67 green = 0.0f;
68 blue = 1.0f;
69 }
70 else // dec blue
71 {
72 red = 1.0f;
73 green = 0.0f;
74 blue = 1.0f - ((fHueAmount - 300.0f) * fReciprical60);
75 }
76
77 // Now RGB is set, apply saturation
78 float fSaturationScaleR = 1.0f - red;
79 float fSaturationScaleG = 1.0f - green;
80 float fSaturationScaleB = 1.0f - blue;
81 red += fSaturationScaleR * fSaturation;
82 green += fSaturationScaleG * fSaturation;
83 blue += fSaturationScaleB * fSaturation;
84 clamp(red, 0.0f, 1.0f);
85 clamp(green, 0.0f, 1.0f);
86 clamp(blue, 0.0f, 1.0f);
87
88 // Now saturation is added, apply brightness
89 float fBrightnessInv = 1.0f - fBrightness;
90 red -= fBrightnessInv;
91 green -= fBrightnessInv;
92 blue -= fBrightnessInv;
93 clamp(red, 0.0f, 1.0f);
94 clamp(green, 0.0f, 1.0f);
95 clamp(blue, 0.0f, 1.0f);
96 }

98 void CColourf::getHSB(float& fHue, float& fSaturation, float& fBrightness) const
99 {
100 // Get maximum and minimum values of current RGB values
101 float cmax = std::max(red, std::max(green, blue));
102 float cmin = std::min(red, std::min(green, blue));
103 float diff = cmax - cmin; // diff of cmax and cmin.
104 fHue = -1, fSaturation = -1;
105
106 // If cmax and cmax are equal then fHue = 0
107 if (cmax == cmin)
108 fHue = 0;
109
110 // If cmax equal r then compute fHue
111 else if (cmax == red)
112 fHue = float(fmod(60 * ((green - blue) / diff) + 360, 360));
113
114 // If cmax equal g then compute fHue
115 else if (cmax == green)
116 fHue = float(fmod(60 * ((blue - red) / diff) + 120, 360));
117
118 // If cmax equal b then compute fHue
119 else if (cmax == blue)
120 fHue = float(fmod(60 * ((red - green) / diff) + 240, 360));
121
122 fHue /= 360.0f;
123
124 // If cmax equal zero
125 if (cmax == 0)
126 fSaturation = 0;
127 else
128 fSaturation = (diff / cmax);// *100;
129
130 // Compute fBrightness
131 fBrightness = cmax;// *100;
132 }

135 {
136 clamp(fValue, 0.0f, 1.0f);
137
138 CColourf colour;
139 float fDiff = red - other.red; // -1.0f (This is 0.0f and other is 1.0f) to 1.0f (This is 1.0f and other is 0.0f)
140 fDiff *= -1.0f; // 1.0f (This is 0.0f and other is 1.0f) to -1.0f (This is 1.0f and other is 0.0f)
141 colour.red = red + (fDiff * fValue);
142
143 fDiff = green - other.green;
144 fDiff *= -1.0f;
145 colour.green = green + (fDiff * fValue);
146
147 fDiff = blue - other.blue;
148 fDiff *= -1.0f;

151 fDiff = alpha - other.alpha;
152 fDiff *= -1.0f;
153 colour.alpha = alpha + (fDiff * fValue);
