21 void CProfiler::begin(const std::string& strSectionName)
22 {
23 // Add named section if it doesn't exist.
24 auto it = _mmapSections.find(strSectionName);
25 if (_mmapSections.end() == it)
26 {
27 SSection newSection;
28 newSection.dAccumulatedTimeInSeconds = 0;
29 newSection.iBeginCalledCount = 0;
30 _mmapSections[strSectionName] = newSection;
31 it = _mmapSections.find(strSectionName);
32 }
33
34 // If this section is "main", then reset all sections
35 if ("main" == strSectionName)
36 {
37 auto it2 = _mmapSections.begin();
38 while (it2 != _mmapSections.end())
39 {
40 it2->second.dAccumulatedTimeInSeconds = 0;
41 it2->second.iBeginCalledCount = 0;
42 it2++;
43 }
44 }
45
46 it->second.timer.update();
47 it->second.iBeginCalledCount++;
48 if (it->second.iBeginCalledCount == 1)
49 it->second.dAccumulatedTimeInSeconds = 0;
50 }

98
99 // Create a vector of pairs
100 std::vector<std::pair<std::string, CProfiler::SSection>> vec;
101
102 // Copy key-value pairs from the map to the vector
103 std::map<std::string, CProfiler::SSection> ::iterator it;
104 for (it = _mmapSections.begin(); it != _mmapSections.end(); it++)
105 {
106 vec.push_back(make_pair(it->first, it->second));
107 }
108
109 // Sort the vector by increasing order of its pair's second value's dAccumulatedTimeInSeconds value.
110 sort(vec.begin(), vec.end(), sortByVal);
111
112 // Find main and get it's accumulated time
113 auto itMain = _mmapSections.find("main");
114 ThrowIfTrue(_mmapSections.end() == itMain, "CProfiler::getResults() failed. The \"main\" section doesn't exist.");
115 double dMainTime = itMain->second.dAccumulatedTimeInSeconds;
116
117 // Prevent divide by zero error below.
118 if (dMainTime <= 0)
119 dMainTime = 0.00001;
120
121 // Now copy vector to vector and return
122 for (size_t i = 0; i < vec.size(); i++)
123 {
124 SProfilerResults results;
125 results.dAccumulatedTimeSeconds = vec[i].second.dAccumulatedTimeInSeconds;
126 results.fPercentageOfMain = float(results.dAccumulatedTimeSeconds / dMainTime);
127 results.fPercentageOfMain *= 100;
128 results.strSectionName = vec[i].first;
129 vResults.push_back(results);
130 }
