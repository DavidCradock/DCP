72 bool CImage::load(const std::string& strFilename, bool bFlipForOpenGL)
73 {
74 free();
75 stbi_set_flip_vertically_on_load(bFlipForOpenGL);
76
77 // Get number of channels in the image file
78 int iDims[2];
79 int iNumChannels = 3;
80 loadInfo(strFilename, iDims[0], iDims[1], iNumChannels);
81 stbi_uc* pixels = 0;
82 if (4 == iNumChannels)
83 pixels = stbi_load(strFilename.c_str(), &_miWidth, &_miHeight, &_miNumChannels, STBI_rgb_alpha);
84 else if (3 == iNumChannels)
85 pixels = stbi_load(strFilename.c_str(), &_miWidth, &_miHeight, &_miNumChannels, STBI_rgb);
86 else if (1 == iNumChannels)
87 pixels = stbi_load(strFilename.c_str(), &_miWidth, &_miHeight, &_miNumChannels, 1);
88
89 if (!pixels)
90 return false;
91
92 // If number of channels is 1, then we convert that 1 channel to 3 and duplicate the R to G and B
93 if (1 == iNumChannels)
94 {
95 _miNumChannels = 3;
96 }
97
98 // Compute size and allocate
99 _muiDataSize = _miWidth * _miHeight * _miNumChannels;
100 _mpData = new unsigned char[_muiDataSize];
101
102 if (1 != iNumChannels)
103 memcpy(_mpData, pixels, static_cast<size_t>(_muiDataSize));
104 else // We need to copy the R to G and B
105 {
106 unsigned int iPixelIndex = 0;
107 for (unsigned int i = 0; i < _muiDataSize; i += 3)
108 {
109 _mpData[i] = pixels[iPixelIndex];
110 _mpData[i + 1] = pixels[iPixelIndex];
111 _mpData[i + 2] = pixels[iPixelIndex];
112 iPixelIndex++;
113 }
114 }
115 stbi_image_free(pixels);
116 return true;
117 }

253 void CImage::flipVertically(void)
254 {
255 ThrowIfTrue(!_mpData, "CImage::flipVertically() failed. Image not yet created.");
256
257 // Size of a row
258 unsigned int iRowSize = _miWidth * _miNumChannels;
259
260 // Allocate new flipped image
261 unsigned char* pNewImageStartAddress = new unsigned char[_muiDataSize];
262 unsigned char* pNewImage = pNewImageStartAddress;
263 ThrowIfTrue(0 == pNewImage, "CImage::flipVertically() failed to allocate memory.");
264
265 // Get pointer to current image
266 unsigned char* pOldImage = _mpData;
267 // Increment old image pointer to point to last row
268 pOldImage += iRowSize * (_miHeight - 1);
269
270 // Copy each row into new image
271 unsigned int iRowSizeBytes = iRowSize * sizeof(unsigned char);
272 for (int iRow = 0; iRow < _miHeight; ++iRow)
273 {
274 memcpy(pNewImage, pOldImage, iRowSizeBytes);
275 // Adjust pointers
276 pNewImage += iRowSizeBytes;
277 pOldImage -= iRowSizeBytes;
278 }
279 // Now pNewImage contains flipped image data
280 delete[] _mpData; // Delete old image data
281 _mpData = pNewImageStartAddress; // Make image data point to the new data
282 }

439 void CImage::copyRectTo(CImage& destImage, int iSrcPosX, int iSrcPosY, int iSrcWidth, int iSrcHeight, int iDestPosX, int iDestPosY) const
440 {
441 // Check that both images have data
442 ThrowIfTrue(!_mpData, "CImage::copyRectTo() failed. Source image not yet created.");
443 ThrowIfTrue(!destImage._mpData, "CImage::copyRectTo() failed. Destination image not yet created.");
444
445 // Compute source rect
446 int iSrcLeft = iSrcPosX;
447 int iSrcBot = iSrcPosY;
448 int iSrcRight = iSrcLeft + iSrcWidth;
449 int iSrcTop = iSrcBot + iSrcHeight;
450 // Compute destination rect
451 int iDstLeft = iDestPosX;
452 int iDstBot = iDestPosY;
453 int iDstRight = iDstLeft + iSrcWidth;
454 int iDstTop = iDstBot + iSrcHeight;
455
456 // The above may be invalid due to different sizes, invalid positions, dims etc.
457 // Invalid starting positions
458 if (iSrcLeft >= _miWidth)
459 return;
460 if (iSrcBot >= _miHeight)
461 return;
462 if (iDstLeft >= destImage._miWidth)
463 return;
464 if (iDstBot >= destImage._miHeight)
465 return;
466 // Clamp right and top to edges of their respective images
467 clamp(iSrcRight, iSrcLeft, _miWidth);
468 clamp(iSrcTop, iSrcBot, _miHeight);
469 clamp(iDstRight, iDstLeft, destImage._miWidth);
470 clamp(iDstTop, iDstBot, destImage._miHeight);
471 // Compute rect dims for both images
472 unsigned int iSrcRectWidth = iSrcRight - iSrcLeft;
473 unsigned int iSrcRectHeight = iSrcTop - iSrcBot;
474 unsigned int iDstRectWidth = iDstRight - iDstLeft;
475 unsigned int iDstRectHeight = iDstTop - iDstBot;
476 // Compute smallest rect
477 unsigned int iMinWidth = iSrcRectWidth;
478 if (iMinWidth > iDstRectWidth)
479 iMinWidth = iDstRectWidth;
480 unsigned int iMinHeight = iSrcRectHeight;
481 if (iMinHeight > iDstRectHeight)
482 iMinHeight = iDstRectHeight;
483 // If minimum = zero, then do nothing
484 if (iMinWidth == 0)
485 return;
486 if (iMinHeight == 0)
487 return;
488
489 unsigned char colTmp[4];
490 unsigned int isx, isy;
491 unsigned int idx, idy;
492 for (unsigned int ix = 0; ix < iMinWidth; ++ix)
493 {
494 for (unsigned int iy = 0; iy < iMinHeight; ++iy)
495 {
496 isx = iSrcLeft + ix;
497 isy = iSrcBot + iy;
498 idx = iDstLeft + ix;
499 idy = iDstBot + iy;
500 getPixel(isx, isy, colTmp[0], colTmp[1], colTmp[2], colTmp[3]);
501 destImage.setPixel(idx, idy, colTmp[0], colTmp[1], colTmp[2], colTmp[3]);
502 }
503 }
504 }

506 void CImage::copyToAddBorder(CImage& outputImage) const
507 {
508 ThrowIfTrue(!_mpData, "CImage::copyToAddBorder() failed. Image data doesn't exist.");
509
510 // Compute new larger dimensions and create the larger image
511 int newWidth = _miWidth + 2;
512 int newHeight = _miHeight + 2;
513 outputImage.createBlank(newWidth, newHeight, _miNumChannels);
514
515 // Copy this image to the centre of the larger image
516 copyRectTo(outputImage, 0, 0, _miWidth, _miHeight, 1, 1);
517
518 // Now copy the edges of this image to the destination image
519 unsigned char r, g, b, a;
520 int heightOfOutputImageMinusOne = newHeight - 1;
521 // Top and bottom edges
522 for (int iX = 0; iX < _miWidth; iX++)
523 {
524 // Top pixel row
525 getPixel(iX, 0, r, g, b, a);
526 outputImage.setPixel(iX + 1, 0, r, g, b, a);
527
528 // Bottom pixel row
529 getPixel(iX, _miHeight - 1, r, g, b, a);
530 outputImage.setPixel(iX + 1, heightOfOutputImageMinusOne, r, g, b, a);
531 }
532 int widthOfOutputImageMinusOne = newWidth - 1;
533 // Left and right edges
534 for (int iY = 0; iY < _miHeight; iY++)
535 {
536 // Left pixel column
537 getPixel(0, iY, r, g, b, a);
538 outputImage.setPixel(0, iY + 1, r, g, b, a);
539
540 // Right pixel column
541 getPixel(_miWidth - 1, iY, r, g, b, a);
542 outputImage.setPixel(widthOfOutputImageMinusOne, iY + 1, r, g, b, a);
543 }

636 void CImage::normalmap(CImage& outputImage, float fScale) const
637 {
638 ThrowIfTrue(!_mpData, "CImage::normalmap() failed. Image data doesn't exist.");
639
640 clamp(fScale, 0.0f, 1.0f);
641
642 // Copy this image into a new one so this is left unaffected.
643 // This uses the copyToAddBorder() method which adds a border and copies the edge pixels to the new pixels in the border.
644 // This makes it so we don't have to mess around with edge cases.
645 CImage imageGreyscale;
646 copyToAddBorder(imageGreyscale);
647
648 // Greyscale the image
649 imageGreyscale.greyscaleSimple();
650
651 // Create output image with the same size as this one
652 outputImage.createBlank(_miWidth, _miHeight, 3);
653
654 // Now loop through greyscale image, computing each normal and storing in the output image.
655 unsigned char r[3], g[3], b[3], a;
656 float fX, fY, fZ;
657 float fLength;
658 for (int y = 0; y < _miHeight; y++)
659 {
660 for (int ix = 0; ix < _miWidth; ix++)
661 {
662 // we add +1 to imageGreyscale pixel positions as it has a border
663
664 // Get height values of centre and surrounding pixels
665 imageGreyscale.getPixel(ix + 1, y + 1, r[0], g[0], b[0], a); // Current pixel
666 imageGreyscale.getPixel(ix, y + 1, r[1], g[1], b[1], a); // Left pixel
667 imageGreyscale.getPixel(ix + 1, y + 2, r[2], g[2], b[2], a); // Above pixel
668
669 fX = float(r[1] - r[0]) / 255.0f; // Convert to -1.0f to 1.0f
670 fY = float(r[2] - r[0]) / 255.0f; // ....
671 fZ = fScale;
672
673 // Compute length of vector and normalize
674 fLength = sqrt((fX * fX) + (fY * fY) + (fZ * fZ));
675 if (areFloatsEqual(fLength, 0.0f)) // If length is nearly zero, just set as up vector
676 {
677 fX = 0.0f;
678 fY = 0.0f;
679 fZ = fScale;
680 }
681 else
682 {
683 fX = fX / fLength;
684 fY = fY / fLength;
685 fZ = fZ / fLength;
686 }
687
688 // Convert from -1, +1 to 0, 255
689 fX += 1.0f; fX *= 127.0f;
690 fY += 1.0f; fY *= 127.0f;
691 fZ += 1.0f; fZ *= 127.0f;
692 r[0] = unsigned char(fX);
693 g[0] = unsigned char(fY);
694 b[0] = unsigned char(fZ);
695 outputImage.setPixel(ix, y, r[0], g[0], b[0], a);
696 }
