55 void CResourceGPUProgramsOpenGL::setProgramSourceFromFile(const std::string& strVertexProgramSourceFilename, const std::string& strFragmentProgramSourceFilename)
56 {
57 // Load in source code
58 prim->strVertexSourceCode.clear();
59 prim->strFragmentSourceCode.clear();
60 std::ifstream vShaderFile;
61 std::ifstream fShaderFile;
62 // ensure ifstream objects can throw exceptions:
63 vShaderFile.exceptions(std::ifstream::failbit | std::ifstream::badbit);
64 fShaderFile.exceptions(std::ifstream::failbit | std::ifstream::badbit);
65 try
66 {
67 vShaderFile.open(strVertexProgramSourceFilename);
68 fShaderFile.open(strFragmentProgramSourceFilename);
69 std::stringstream vShaderStream, fShaderStream;
70 // Read file's buffer contents into streams
71 vShaderStream << vShaderFile.rdbuf();
72 fShaderStream << fShaderFile.rdbuf();
73 vShaderFile.close();
74 fShaderFile.close();
75 // convert stream into string
76 prim->strVertexSourceCode = vShaderStream.str();
77 prim->strFragmentSourceCode = fShaderStream.str();
78 }
79 catch (std::ifstream::failure e)
80 {
81 ThrowIfTrue(1, "CResourceGPUProgramsOpenGL::setProgramSourceFromFile() failed to read in program code for either " + strVertexProgramSourceFilename + " or " + strFragmentProgramSourceFilename);
82 }
83
84 }

96 void CResourceGPUProgramsOpenGL::upload(void)
97 {
98 // Make sure the source code as text exists
99 ThrowIfTrue(0 == prim->strVertexSourceCode.length(), "CResourceGPUProgramsOpenGL::upload() failed. The vertex program's source code hasn't been set. Did you forget to call setProgramSourceFromFile()?");
100 ThrowIfTrue(0 == prim->strFragmentSourceCode.length(), "CResourceGPUProgramsOpenGL::upload() failed. The fragment program's source code hasn't been set. Did you forget to call setProgramSourceFromFile()?");;
101
102 // Free the linked shader programs if they already exist
103 free();
104
105 const char* vShaderCode = prim->strVertexSourceCode.c_str();
106 const char* fShaderCode = prim->strFragmentSourceCode.c_str();
107
108 // Attempt to compile shaders
109 unsigned int vertex, fragment;
110 int success;
111 char infoLog[1024];
112
113 // Vertex Shader
114 vertex = glCreateShader(GL_VERTEX_SHADER);
115 glShaderSource(vertex, 1, &vShaderCode, NULL);
116 glCompileShader(vertex);
117 glGetShaderiv(vertex, GL_COMPILE_STATUS, &success);
118 if (!success)
119 {
120 glGetShaderInfoLog(vertex, 1024, NULL, infoLog);
121 std::string err("CResourceGPUProgramsOpenGL::upload() failed to compile the vertex program. ");
122 err.append(infoLog);
123 Throw(err);
124 };
125
126 // Fragment Shader
127 fragment = glCreateShader(GL_FRAGMENT_SHADER);
128 glShaderSource(fragment, 1, &fShaderCode, NULL);
129 glCompileShader(fragment);
130 glGetShaderiv(fragment, GL_COMPILE_STATUS, &success);
131 if (!success)
132 {
133 glGetShaderInfoLog(fragment, 1024, NULL, infoLog);
134 std::string err("CResourceGPUProgramsOpenGL::upload() failed to compile the fragment program. ");
135 err.append(infoLog);
136 Throw(err);
137 };
138
139 // Shader Program
140 prim->uiProgramID = glCreateProgram();
141 glAttachShader(prim->uiProgramID, vertex);
142 glAttachShader(prim->uiProgramID, fragment);
143 glLinkProgram(prim->uiProgramID);
144
145 // Show linking errors if any
146 glGetProgramiv(prim->uiProgramID, GL_LINK_STATUS, &success);
147 if (!success)
148 {
149 std::string err("CResourceGPUProgramsOpenGL::upload() failed to link the programs. ");
150 err.append(infoLog);
151 Throw(err);
152 }
153
154 // Delete the shaders as they're linked into our program now and no longer necessary
155 glDeleteShader(vertex);
156 glDeleteShader(fragment);
157 }
