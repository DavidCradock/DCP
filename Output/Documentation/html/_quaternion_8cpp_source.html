190 void CQuaternion::setFromEuler(float fAngleRadiansX, float fAngleRadiansY, float fAngleRadiansZ)
191 {
192 double dSY = sin(fAngleRadiansZ * 0.5f);
193 double dSP = sin(fAngleRadiansY * 0.5f);
194 double dSR = sin(fAngleRadiansX * 0.5f);
195 double dCY = cos(fAngleRadiansZ * 0.5f);
196 double dCP = cos(fAngleRadiansY * 0.5f);
197 double dCR = cos(fAngleRadiansX * 0.5f);
198
199 q[0] = float(dSR * dCP * dCY - dCR * dSP * dSY);
200 q[1] = float(dCR * dSP * dCY + dSR * dCP * dSY);
201 q[2] = float(dCR * dCP * dSY - dSR * dSP * dCY);
202 q[3] = float(dCR * dCP * dCY + dSR * dSP * dSY);
203 normalise();
204 }

206 CVector3f CQuaternion::getEuler(void) const
207 {
208 float fSq[4];
209 fSq[0] = q[0] * q[0];
210 fSq[1] = q[1] * q[1];
211 fSq[2] = q[2] * q[2];
212 fSq[3] = q[3] * q[3];
213
214 double m11 = fSq[3] + fSq[0] - fSq[1] - fSq[2];
215 double m21 = 2 * (q[0] * q[1] + q[2] * q[3]);
216 double m31 = 2 * (q[2] * q[0] - q[1] * q[3]);
217 double m32 = 2 * (q[2] * q[1] + q[0] * q[3]);
218 double m33 = fSq[3] - fSq[0] - fSq[1] + fSq[2];
219 return CVector3f(float(atan2(m32, m33)), asinf(float(-m31)), float(atan2(m21, m11)));
220 }

222 void CQuaternion::setFromMatrix(const CMatrix& m)
223 {
224 float fTr = m.m[0] + m.m[5] + m.m[9] + 1.0f;
225 // If the trace of the matrix is greater than 0, perform an instant calculation
226 if (fTr > 0)
227 {
228 float fS = 0.5f / sqrtf(fTr);
229
230 q[3] = 0.25f / fS;
231 q[0] = (m.m[9] - m.m[6]) * fS;
232 q[1] = (m.m[2] - m.m[8]) * fS;
233 q[2] = (m.m[4] - m.m[1]) * fS;
234 }
235 // if the first diagonal element is the largest
236 else if (m.m[0] > m.m[5] && m.m[0] > m.m[10])
237 {
238 float fS = float(1 / (sqrt(m.m[0] - m.m[5] - m.m[10] + 1.0f) * 2));
239 q[0] = 0.5f * fS;
240 q[1] = (m.m[1] + m.m[4]) * fS;
241 q[2] = (m.m[2] + m.m[8]) * fS;
242 q[3] = (m.m[6] + m.m[9]) * fS;
243 }
244 // if the second.....
245 else if (m.m[5] > m.m[10])
246 {
247 float fS = float(1 / (sqrt(m.m[5] - m.m[0] - m.m[10] + 1.0f) * 2));
248 q[0] = (m.m[1] + m.m[4]) * fS;
249 q[1] = 0.5f * fS;
250 q[2] = (m.m[6] + m.m[9]) * fS;
251 q[3] = (m.m[2] + m.m[8]) * fS;
252 }
253 // if the last element is the largest
254 else
255 {
256 float fS = float(1 / (sqrt(m.m[10] - m.m[0] - m.m[5] + 1.0f) * 2));
257 q[0] = (m.m[2] + m.m[8]) * fS;
258 q[1] = (m.m[6] + m.m[9]) * fS;
259 q[2] = 0.5f * fS;
260 q[3] = (m.m[1] + m.m[4]) * fS;
261 }

270 CQuaternion CQuaternion::getSLERP(const CQuaternion& quat, float fInterval) const
271 {
272 CQuaternion q(quat);
273
274 float fDot = q[0] * q.q[0] + q[1] * q.q[1] + q[2] * q.q[2] + q[3] * q.q[3];
275 if (fDot < 0.0f)
276 {
277 q = q.getNegative();
278 fDot = -fDot;
279 }
280
281 // Inaccurate, use lerp instead
282 if (fDot < 1.00001f && fDot > 0.99999f)
283 {
284 return getLERP(q, fInterval);
285 }
286
287 // Calculate the angle between the quaternions
288 float fTheta = acosf(fDot);
289
290 return ((*this * sinf(fTheta * (1 - fInterval)) + q * sinf(fTheta * fInterval)) / sinf(fTheta));
291 }
