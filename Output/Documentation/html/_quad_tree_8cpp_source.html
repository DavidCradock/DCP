152 void CQuadTree::addEntity(const std::string& strName, int iPosX, int iPosY, int iUserData, void* pUserData)
153 {
154 // Make sure the entity doesn't already exist by checking the hashmap
155 std::map<std::string, CQuadTreeEntity*>::iterator it = _mmapEntities.find(strName);
156 ThrowIfTrue(_mmapEntities.end() != it, "CQuadTree::addEntity() failed. The entity name of " + strName + " already exists.");
157
158 // Create new entity, setting it's owner to 0
159 CQuadTreeEntity* pEntity = new CQuadTreeEntity(strName, iPosX, iPosY, 0, iUserData, pUserData);
160 ThrowIfFalse(pEntity, "CQuadTree::addEntity() failed to allocate memory for new entity.");
161
162 // Add entity to hashmap for fast lookup
163 _mmapEntities[strName] = pEntity;
164
165 // Determine whether the given position of the entity fits in the root node
166 if (!_mpRootNode->_mRectRegion.doesPositionFitWithin(iPosX, iPosY))
167 {
168 // The position of the new entity doesn't fit within the root node's area
169 // We're going to have to recreate the entire tree
170
171 // Delete root node, which will delete all children and their children and so on.
172 // This does not delete the entities (As they're stored in the _mmapEntities hashmap)
173 // But before deleting the root node, get it's currently set region which we will multiply to get
174 // the root node's new dimensions
175 CRect rectOldRootNodeRegion = _mpRootNode->_mRectRegion;
176 delete _mpRootNode;
177
178 // Inscrease the old root node's region size until the new entity's position fits within
179 bool bNewPositionFits = false;
180 while (!bNewPositionFits)
181 {
182 rectOldRootNodeRegion.resizeArea(_miRectSizeIncreaseMultiplier);
183 bNewPositionFits = rectOldRootNodeRegion.doesPositionFitWithin(iPosX, iPosY);
184 }
185
186 // Now re-create the root node, passing 0 as the pointer to the parent node and the new region size
187 _mpRootNode = new CQuadTreeNode(rectOldRootNodeRegion, 0, this);
188 ThrowIfFalse(_mpRootNode, "CQuadTree::addEntity() failed. Unable to allocate memory for new root node.");
189
190 // Now re-insert all entities stored in the hash map back into the tree
191 // The new entity has already been added to the hash map, so no need to insert it seperately
192 it = _mmapEntities.begin();
193 while (it != _mmapEntities.end())
194 {
195 _mpRootNode->addEntity(it->second);
196 it++;
197 }
198 }
199 else // The entity position does fit inside the rect of this node, simply add it
200 {
201 _mpRootNode->addEntity(pEntity);
202 }
203 }

205 void CQuadTree::removeEntity(const std::string& strName)
206 {
207 // Make sure the entity exists by checking the hashmap
208 std::map<std::string, CQuadTreeEntity*>::iterator it = _mmapEntities.find(strName);
209 ThrowIfTrue(_mmapEntities.end() == it, "CQuadTree::removeEntity() failed. The entity name of " + strName + " doesn't exist.");
210
211 // Get the node the entity is stored in to remove the entity from itself.
212 CQuadTreeNode* pNodeContainingRemovedEntity = it->second->_mpNodeOwner;
213 pNodeContainingRemovedEntity->removeEntity(it->second);
214
215 // Remove the entity from the hashmap and delete it
216 CQuadTreeEntity* pEntity = it->second;
217 _mmapEntities.erase(it);
218 delete pEntity;
219
220 // Now check to see if the node is now empty and if so, remove it
221 if (!pNodeContainingRemovedEntity->hasEntitiesInThisAndAllChildren())
222 {
223 // Is the node which held the removed entity, the root node
224 bool bNodeIsRoot = _mpRootNode == pNodeContainingRemovedEntity;
225
226 if (!bNodeIsRoot) // The node wasn't the root node and therefore pNodeParent will contain a valid pointer to it's parent node.
227 {
228 // Store pointer to the parent
229 CQuadTreeNode* pNodeParent = pNodeContainingRemovedEntity->_mpParentNode;
230
231 // We're done, except...
232 // We need to traverse up to the root node and delete any NOW empty nodes, which may
233 // have only existed to hold the now deleted node.
234 // We also have to remove any parent node's _mpChildNode[4] pointers if we delete a node.
235 while (pNodeParent)
236 {
237 // Go through each of the four possible child nodes of the parent
238 for (int i = 0; i < 4; i++)
239 {
240 if (pNodeParent->_mpChildNode[i]) // There is a child at this array position
241 {
242 if (!pNodeParent->_mpChildNode[i]->hasEntitiesInThisAndAllChildren()) // Node is empty
243 {
244 delete pNodeParent->_mpChildNode[i];
245 pNodeParent->_mpChildNode[i] = 0;
246 }
247 }
248 }
249 pNodeParent = pNodeParent->_mpParentNode;
250 }
251
252 // This will delete the node and all it's children.
253 // No need to delete, as it's deleted above when deleting children.
254 // If we attempted to delete again, all hell breaks loose!
255// delete pNodeContainingRemovedEntity;
256 }
257 }
258 // The node which contained the entity has other entities, leave it alone.

268 void CQuadTree::removeAllEntities(bool bResetTree)
269 {
270 // Go through each entity, asking each node which it's in to remove itself
271 std::map<std::string, CQuadTreeEntity*>::iterator it = _mmapEntities.begin();
272 while (it != _mmapEntities.end())
273 {
274 // Get the node the entity is stored in to remove the entity from itself.
275 it->second->_mpNodeOwner->removeEntity(it->second);
276
277 // Remove the entity from the hashmap and delete it
278 CQuadTreeEntity* pEntity = it->second;
279 _mmapEntities.erase(it);
280 delete pEntity;
281 it = _mmapEntities.begin();
282 }
283
284 // Now all entities are removed and deleted, reset the tree if bResetTree desires it
285 if (bResetTree)
286 {
287 // Get the root node's rect, so we can re-create it
288 CRect rectRootNode = _mpRootNode->_mRectRegion;
289 delete _mpRootNode;
290 _mpRootNode = new CQuadTreeNode(rectRootNode, 0, this);
291
292 // Reset current max node depth
293 _muiCurrentMaxNodeDepth = 0;
294
295 // Compute new maximum node depth
296 _computeMaxNodeDepth();
297 }

300 void CQuadTree::setEntityPosition(const std::string& strName, int iNewPosX, int iNewPosY)
301 {
302 // First make sure the named entity exists
303 std::map<std::string, CQuadTreeEntity*>::iterator it = _mmapEntities.find(strName);
304 ThrowIfTrue(it == _mmapEntities.end(), "CQuadTree::setEntityPosition() failed. The named entity of " + strName + " doesn't exist.");
305
306 // First check to see if the new entity position still fits within it's current node
307 // If it does, we simply update the position
308 if (it->second->_mpNodeOwner->_mRectRegion.doesPositionFitWithin(iNewPosX, iNewPosY))
309 {
310 it->second->_miPosX = iNewPosX;
311 it->second->_miPosY = iNewPosY;
312 return;
313 }
314
315 // If we get here, the new position doesn't fit within the entity's current node
316
317 // Remove the entity from the tree and then re-insert it
318 int iUserData = it->second->miUserData;
319 void* pUserData = it->second->mpUserData;
320 removeEntity(strName);
321 addEntity(strName, iNewPosX, iNewPosY, iUserData, pUserData);
322 }

333 void CQuadTree::_computeMaxNodeDepth(void)
334 {
335 // Obtain smallest dimension of root node
336 int iDimOfRootX = _mpRootNode->_mRectRegion.miMaxX - _mpRootNode->_mRectRegion.miMinX;
337 int iDimOfRootY = _mpRootNode->_mRectRegion.miMaxY - _mpRootNode->_mRectRegion.miMinY;
338 int iSmallestDim = iDimOfRootX;
339 if (iDimOfRootY < iDimOfRootX)
340 iSmallestDim = iDimOfRootY;
341 // Now compute max node depth
342 _muiMaxNodeDepth = 0;
343 while (iSmallestDim > 2)
344 {
345 iSmallestDim /= 2;
346 _muiMaxNodeDepth++;
347 }
348 }
