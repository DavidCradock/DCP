533/* Note that memset() is a byte assignment and this is a 32-bit assignment, so they're not directly equivalent. */
534SDL_FORCE_INLINE void SDL_memset4(void *dst, Uint32 val, size_t dwords)
535{
536#if defined(__GNUC__) && defined(__i386__)
537 int u0, u1, u2;
538 __asm__ __volatile__ (
539 "cld \n\t"
540 "rep ; stosl \n\t"
541 : "=&D" (u0), "=&a" (u1), "=&c" (u2)
542 : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, dwords))
543 : "memory"
544 );
545#else
546 size_t _n = (dwords + 3) / 4;
547 Uint32 *_p = SDL_static_cast(Uint32 *, dst);
548 Uint32 _val = (val);
549 if (dwords == 0) {
550 return;
551 }
552 switch (dwords % 4) {
553 case 0: do { *_p++ = _val; SDL_FALLTHROUGH;
554 case 3: *_p++ = _val; SDL_FALLTHROUGH;
555 case 2: *_p++ = _val; SDL_FALLTHROUGH;
556 case 1: *_p++ = _val;
557 } while ( --_n );
558 }
559#endif
560}

780/**
781 * If a * b would overflow, return -1. Otherwise store a * b via ret
782 * and return 0.
783 *
784 * \since This function is available since SDL 2.24.0.
785 */
786SDL_FORCE_INLINE int SDL_size_mul_overflow (size_t a,
787 size_t b,
788 size_t *ret)
789{
790 if (a != 0 && b > SDL_SIZE_MAX / a) {
791 return -1;
792 }
793 *ret = a * b;
794 return 0;
795}

812 * and return 0.
